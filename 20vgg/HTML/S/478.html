 278   /**
 279      @brief calc a mean of each input channel
 280      @param (x) input images
 281      @sa forward
 282      @sa backward
 283      @details this is an auxiliary function called
 284      from forward. mean(i) = average of pixel
 285      values over all pixels of all images in layer i
 286   */
 287   __device__ __host__
 288   vec<IC> mean_bij(array4<maxB,IC,H,W>& x) {
 289     const idx_t B = x.B;
 290     vec<IC> mean;
 291     mean.init_const(IC, 0);
 292     for (idx_t ic = 0; ic < IC; ic++) {
 293       real s = 0.0;
 294       for (idx_t b = 0; b < B; b++) {
 295         for (idx_t i = 0; i < H; i++) {
 296           for (idx_t j = 0; j < W; j++) {
 297             s += x(b,ic,i,j);
 298           }
 299         }
 300       }
 301       mean(ic) = s / (B * H * W);
 302     }
 303     return mean;
 304   }
 305   /**
 306      @brief calc a standard deviation of each input channel
 307      @param (x) input images
 308      @param (mu) mean 
 309      @sa forward
 310      @sa backward
 311      @details this is an auxiliary function called
 312      from forward. inv_std(i) = 1 / sqrt(standard deviation of pixel
 313      values over all pixels of all images in layer i)
 314   */
 315   __device__ __host__
 316   vec<IC>& inv_std_bij(array4<maxB,IC,H,W>& x, vec<IC>& mu) {
 317     const idx_t B = x.B;
 318     const real epsilon = 2.0e-5;
 319     const real l_BHW = 1 / (real)(B * H * W);
 320     inv_std.set_n(IC);
 321     for (idx_t ic = 0; ic < IC; ic++) {
 322       real s = 0.0;
 323       for (idx_t b = 0; b < B; b++) {
 324         for (idx_t i = 0; i < H; i++) {
 325           for (idx_t j = 0; j < W; j++) {
 326             real ds = x(b,ic,i,j) - mu(ic);
 327             s += ds * ds;
 328           }
 329         }
 330       }
 331       inv_std(ic) = 1.0 / sqrt(s * l_BHW + epsilon);
 332     }
 333     return inv_std;
 334   }

 347   __device__ __host__ 
 348   void forward_base(array4<maxB,IC,H,W>& x) {
 349     const idx_t B = x.B;
 350     x_hat.set_n_rows(B);
 351     y.set_n_rows(B);
 352     if (B * H * W > 1) {
 353       vec<IC> mu = mean_bij(x);
 354       inv_std = inv_std_bij(x, mu);
 355       for (idx_t b = 0; b < B; b++) {
 356         for (idx_t ic = 0; ic < IC; ic++) {
 357           for (idx_t i = 0; i < H; i++) {
 358             for (idx_t j = 0; j < W; j++) {
 359               x_hat(b,ic,i,j) = (x(b,ic,i,j) - mu(ic)) * inv_std(ic);
 360               y(b,ic,i,j) = gamma(ic) * x_hat(b,ic,i,j) + beta(ic);
 361             }
 362           }
 363         }
 364       }
 365     } else {
 366       for (idx_t b = 0; b < B; b++) {
 367         for (idx_t ic = 0; ic < IC; ic++) {
 368           for (idx_t i = 0; i < H; i++) {
 369             for (idx_t j = 0; j < W; j++) {
 370               y(b,ic,i,j) = x(b,ic,i,j);
 371             }
 372           }
 373         }
 374       }
 375     }
 376   }

 446   /**
 447      @brief the baseline (serial) implementation of backward
 448      called both by cpu implementation (backward_cpu) and 
 449      gpu implementation (backward_dev). the call sequence
 450      backward -> backward_cpu -> backward_base on cpu and
 451      and is backward -> backward_gpu -> backward_global -> backward_dev -> backward_base
 452      @param (gy) the gradient of loss wrt y 
 453      @sa backward
 454      @sa backward_gpu
 455      @sa backward_global
 456      @sa backward_dev
 457   */
 458   __device__ __host__
 459   void backward_base(array4<maxB,IC,H,W>& gy) {
 460     const idx_t B = gy.B;
 461     gx.set_n_rows(B);
 462     gbeta.set_n(IC);
 463     ggamma.set_n(IC);
 464     if (B * H * W > 1) {
 465       for (idx_t ic = 0; ic < IC; ic++) {
 466         real s = 0.0, t = 0.0;
 467         for (idx_t b = 0; b < B; b++) {
 468           for (idx_t i = 0; i < H; i++) {
 469             for (idx_t j = 0; j < W; j++) {
 470               s += gy(b,ic,i,j);
 471               t += gy(b,ic,i,j) * x_hat(b,ic,i,j);
 472             }
 473           }
 474         }
 475         gbeta(ic) = s;
 476         ggamma(ic) = t;
 477       }
 478       real l_BHW = 1 / (real)(B * H * W);
 479       for (idx_t ic = 0; ic < IC; ic++) {
 480         real a = gamma(ic) * inv_std(ic);
 481         real gg = ggamma(ic);
 482         real gb = gbeta(ic);
 483         for (idx_t b = 0; b < B; b++) {
 484           for (idx_t i = 0; i < H; i++) {
 485             for (idx_t j = 0; j < W; j++) {
 486               gx(b,ic,i,j) = a * (gy(b,ic,i,j) - l_BHW * (gg * x_hat(b,ic,i,j) + gb));
 487             }
 488           }
 489         }
 490       }
 491     } else {
 492       for (idx_t b = 0; b < B; b++) {
 493         for (idx_t ic = 0; ic < IC; ic++) {
 494           for (idx_t i = 0; i < H; i++) {
 495             for (idx_t j = 0; j < W; j++) {
 496               gx(b,ic,i,j) = gy(b,ic,i,j);
 497             }
 498           }
 499         }
 500       }
 501     }
 502   }

 613 /**
 614    @brief check the gradient computation of a batch normalization layer
 615    @param (opt) command line option
 616    @param (lgr) logger 
 617    @param (rg) random number generator
 618    @param (B) the number of images
 619    @sa batchnormalization_main
 620    @details it first makes a layer object with initial weights W 
 621    and generates an input (x and t).
 622    it then creates two layers whose weights are slightly different
 623    from the original one by dw/2 (i.e., w-dw/2 and w+dw/2), as well as
 624    two inputs slighly different from the original inputs by dx/2
 625    (x-dx/2 and x+dx/2).  it then computes L(w,x), L(x-dw/2,x-dx/2) and
 626    L(w+dw/2,x+dw/2) and check if L(x+dw/2,x+dx/2)-L(x-dw/2,x-dx/2)
 627    is close to L/x dx + L/w dw.  L/x and L/w are obtained
 628    by backward computation. This is essentially checking if
 629    the gradients obtained by backward computation correctly approximates
 630    the diff of the output.
 631 */
 632 template<idx_t maxB,idx_t IC,idx_t H,idx_t W>
 633   static real batchnormalization_grad_check_rand(cmdline_opt opt, logger * lgr, rnd_gen_t& rg, idx_t B) {
     /*   */
 634   /* initialize batch normalization parameters */
 635   BatchNormalization<maxB,IC,H,W> * bn = new BatchNormalization<maxB,IC,H,W>();
 636   bn->init(opt, lgr, rg);
 637   bn->make_dev();
 638   bn->to_dev();
 639   /* make w - dw/2 and w + dw/2 */
 640   BatchNormalization<maxB,IC,H,W> * bn_minus = bn->copy();
 641   BatchNormalization<maxB,IC,H,W> * bn_plus = bn->copy();
 642   /* make coefficients to make the single loss value */
 643   array4<maxB,IC,H,W> * alpha = new array4<maxB,IC,H,W>();
 644   alpha->make_dev(opt.gpu_algo);
 645   alpha->init_uniform(B, rg, 0.0, 1.0);
 646   alpha->to_dev();
 647   /* make input (x) */
 648   array4<maxB,IC,H,W> * x = new array4<maxB,IC,H,W>();
 649   x->make_dev(opt.gpu_algo);
 650   x->init_uniform(B, rg, 0.0, 1.0);
 651   x->to_dev();
 652   /* forward and backward */
 653   array4<maxB,IC,H,W>& y = bn->forward(*x);
 654   array4<maxB,IC,H,W>& gx = bn->backward(*alpha);
 655   /* ensure the gradient is back to host */
 656   bn->to_host();
 657 
 658   /* make dx */
 659   real e = 1.0e-4;
 660   array4<maxB,IC,H,W> * dx = new array4<maxB,IC,H,W>();
 661   dx->init_uniform(B, rg, -e, e);
 662   /* make x - dx/2 and x + dx/2 */
 663   array4<maxB,IC,H,W> * x_minus = new array4<maxB,IC,H,W>(*x);
 664   x_minus->make_dev(opt.gpu_algo);
 665   array4<maxB,IC,H,W> * x_plus  = new array4<maxB,IC,H,W>(*x);
 666   x_plus->make_dev(opt.gpu_algo);
 667   /* update on the host and send the to gpu */
 668   x_minus->update(-0.5, *dx);
 669   x_plus->update( 0.5, *dx);
 670   x_minus->to_dev();
 671   x_plus->to_dev();
 672     
 673   /* set gw to a random vector */
 674   bn_minus->rand_grad(rg, -e, e);
 675   bn_plus->set_grad(*bn_minus);
 676   /* send them to gpu */
 677   bn_minus->to_dev();
 678   bn_plus->to_dev();
 679   /* update weights using gw (update runs on gpu) */
 680   bn_minus->update(-0.5);      /* w -= dw/2 */
 681   bn_plus->update(0.5);        /* w += dw/2 */
 682   /* make y(w-dw,x-dx), y(w+dw,x+dx) */
 683   array4<maxB,IC,H,W>& y_minus = bn_minus->forward(*x_minus);
 684   array4<maxB,IC,H,W>& y_plus  = bn_plus->forward(*x_plus);
 685   /* get the result back to host */
 686   y_minus.to_host();
 687   y_plus.to_host();
 688 
 689   /* get the single loss values */
 690   real L_minus = alpha->dot(y_minus);
 691   real L       = alpha->dot(y);
 692   real L_plus  = alpha->dot(y_plus);
 693   /* various inner products */
 694   real gx_gx = gx.dot(gx);                       /* L/xL/x */
 695   real dx_dx = dx->dot(*dx);                     /* L/xdx */
 696   real gx_dx = gx.dot(*dx);                      /* dxdx */
 697   real gw_gw = bn->gw_dot_gw(*bn);               /* L/wL/w */
 698   real dw_dw = bn_minus->gw_dot_gw(*bn_minus);   /* L/wdw */
 699   real gw_dw = bn->gw_dot_gw(*bn_minus);         /* dwdw */
 700 
 701   real rel_e = show_error(gx_gx, dx_dx, gx_dx, gw_gw, dw_dw, gw_dw, L_minus, L, L_plus);
 702   /* clean up */
 703   bn_minus->del_dev();
 704   bn_plus->del_dev();
 705   alpha->del_dev();
 706   x->del_dev();
 707   dx->del_dev();
 708   x_minus->del_dev();
 709   x_plus->del_dev();
 710   
 711   delete bn_minus;
 712   delete bn_plus;
 713   delete alpha;
 714   delete x;
 715   delete dx;
 716   delete x_minus;
 717   delete x_plus;
 718   return rel_e;
 719 }

 721 /**
 722    @brief entry point of this header file
 723    @param (argc) the number of command line args
 724    @param (argv) command line args
 725    @sa batchnormalization_grad_check_rand
 726    @details if this header file is included from
 727    a main C++ file and define batchnormalization_main to be main
 728    (e.g., with -Dbatchnormalization_main=main), then this
 729    function becomes th main function of the executable.
 730    it calls batchnormalization_grad_check_rand repeatedly to test
 731    the implementation of VGG network.
 732 */
 733 int batchnormalization_main(int argc, char ** argv) {
     /*   */
 734   cmdline_opt opt = parse_args(argc, argv);
 735   const idx_t maxB = MAX_BATCH_SIZE;
 736   const idx_t B = min_i(maxB, opt.batch_sz);
 737   const idx_t IC = 64;
 738   const idx_t H = 32;
 739   const idx_t W = 32;
 740   const int n_checks = opt.iters;
 741   /* logger */
 742   logger lgr;
 743   lgr.start_log(opt);
 744   /* initialize random number generator */
 745   rnd_gen_t rg;
 746   rg.seed(opt.weight_seed);
 747   /* check errors */
 748   real max_e = 0.0;
 749   real sum_e = 0.0;
 750   for (int iter = 0; iter < n_checks; iter++) {
 751     printf("==== %d ====\n", iter);
 752     real e = batchnormalization_grad_check_rand<maxB,IC,H,W>(opt, &lgr, rg, B);
 753     max_e = max_r(max_e, e);
 754     sum_e += e;
 755   }
 756   printf("max relative error = %.9f\n", max_e);
 757   printf("avg relative error = %.9f\n", sum_e / n_checks);
 758   lgr.end_log();
 759   return 0;
 760 }
