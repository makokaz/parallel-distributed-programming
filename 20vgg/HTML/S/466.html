   9 /**
  10    @brief grab a mini batch (B training samples), forward, backward and update.
  11    @return the average loss of the mini batch.
  12  */
  13 template<idx_t maxB,idx_t C0,idx_t H,idx_t W,idx_t K,idx_t S,idx_t C1,idx_t nC>
  14 static real train(VGG<maxB,C0,H,W,K,S,C1,nC> * vgg,
     /*   */
  15                   cifar10_dataset<maxB,C0,H,W>& data, idx_t B, long count) {
  16   vgg->lgr->log(1, "=== train %ld - %ld ===", count, count + B);
  17   if (vgg->opt.single_batch) {
  18     data.set_seed(vgg->opt.sample_seed);
  19   }
  20   data.get_data_train(vgg->x, vgg->t, vgg->idxs, B);
  21   real Lsum = vgg->forward_backward_update(vgg->x, vgg->t, vgg->opt.learnrate);
  22   real L = Lsum / B;
  23   int correct = vgg->log_minibatch(0);
  24   vgg->lgr->log(1, "train accuracy %d / %d = %.3f",
  25                 correct, B, correct / (double)B);
  26   vgg->lgr->log(1, "train loss = %.9f", L);
  27   return L;
  28 }

  30 /**
  31    @brief forward compute B_validate validation samples 
  32    (taking several mini batches if necessary)
  33    @return the average loss of the validation data
  34  */
  35 template<idx_t maxB,idx_t C0,idx_t H,idx_t W,idx_t K,idx_t S,idx_t C1,idx_t nC>
  36 static real validate(VGG<maxB,C0,H,W,K,S,C1,nC> * vgg,
     /*   */
  37                      cifar10_dataset<maxB,C0,H,W>& data, long count) {
  38   if (data.n_validate > 0) {
  39     vgg->lgr->log(1, "=== validate %ld - %ld ===", count, count + data.n_validate);
  40     long read_from = 0;
  41     real Lsum = 0.0;
  42     int correct = 0;
  43     while (read_from < data.n_validate) {
  44       long read_to = min_i(read_from + maxB, data.n_validate);
  45       data.get_data_validate(vgg->x, vgg->t, vgg->idxs, read_from, read_to);
  46       vec<maxB>& y = vgg->forward(vgg->x, vgg->t);
  47       y.to_host();
  48       Lsum += y.sum();
  49       correct += vgg->log_minibatch(read_from);
  50       read_from = read_to; 
  51     }
  52     real L = Lsum / data.n_validate;
  53     vgg->lgr->log(1, "validate accuracy %d / %d = %.3f",
  54                   correct, data.n_validate, correct / (double)data.n_validate);
  55     vgg->lgr->log(1, "validate loss = %.9f", L);
  56     return L;
  57   } else {
  58     return 0.0;
  59   }
  60 }

  69 /**
  70    @brief main function of VGG
  71    @details Train VGG network with data from the file specified by
  72    --cifar_data/-d (default: cifar-10-batches-bin/data_batch_1.bin).
  73    If you want to use only a part of data, you can specify a range
  74    by --start_data and --end_data. e.g., --start
  75    Take a number of samples specified by --batch_sz/-d
  76    samples at a time for training.
  77    Occasionally evaluate the network with validation data.
  78    @return the average loss of the validation data
  79  */
  80 int main(int argc, char ** argv) {
     /*   */
  81   cmdline_opt opt = parse_args(argc, argv);
  82   if (opt.error || opt.help) usage(argv[0]);
  83   const idx_t maxB = MAX_BATCH_SIZE;  /**< max batch size (constant) */
  84   const idx_t B  = opt.batch_sz;      /**< true batch size (<= maxB) */
  85   const idx_t C0 = 3;                 /**< input channels (RGB) */
  86   const idx_t H  = 32;                /**< height of an image */
  87   const idx_t W  = 32;                /**< width of an image */
  88   const idx_t K  = 1;        /**< kernel size (K=1 -> 3x3) */
  89   const idx_t S  = 2;        /**< max pooling stride HxW -> H/SxW/S */
  90   const idx_t C1 = N_FIRST_CHANNELS; /**< channels of the first stage */
  91   const idx_t nC = 10;       /**< number of classes */
  92   assert(B <= maxB);
  93   /* logger */
  94   logger lgr;
  95   lgr.start_log(opt);
  96   /* random number */
  97   rnd_gen_t rg;
  98   rg.seed(opt.weight_seed);
  99   /* build model and initialize weights */
 100   lgr.log(1, "model building starts");
 101   VGG<maxB,C0,H,W,K,S,C1,nC> * vgg = new VGG<maxB,C0,H,W,K,S,C1,nC>();
 102   vgg->init(opt, &lgr, rg);
 103   vgg->make_dev();
 104   vgg->to_dev();
 105   lgr.log(1, "model building ends");
 106   /* load data */
 107   cifar10_dataset<maxB,C0,H,W> data;
 108   data.load(lgr, opt.cifar_data, opt.partial_data,
 109             opt.partial_data_seed, opt.validate_ratio,
 110             opt.cifar_data_dump);
 111   data.set_seed(opt.sample_seed);
 112   /* training loop */
 113   long n_trained = 0;
 114   long n_validated = 0;
 115   lgr.log(1, "training starts");
 116   for (long i = 0; i < opt.iters; i++) {
 117     /* train with a mini-batch */
 118     real train_loss = train(vgg, data, B, n_trained);
 119     (void)train_loss;
 120     n_trained += B;
 121     /* evaluate with validation data */
 122     if (data.n_validate > 0 &&
 123         n_trained >= opt.validate_interval * (n_validated + data.n_validate)) {
 124       real validate_loss = validate(vgg, data, n_validated);
 125       (void)validate_loss;
 126       n_validated += data.n_validate;
 127     }
 128   }
 129   lgr.log(1, "training ends");
 130   lgr.end_log();
 131   return 0;
 132 }
