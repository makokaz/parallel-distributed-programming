 169   /**
 170      @brief compute log(softmax(x))
 171      @param (x) a matrix 
 172      @details for 1D vector x x = (x_0, ..., x_{n-1}), 
 173      
 174                            (exp(x_0)     / Σ_j exp(x_j))
 175      logsoftmax(x)_i = log (exp(x_1)     / Σ_j exp(x_j))
 176                            (   ...       / Σ_j exp(x_j))
 177                            (exp(x_{n-1}) / Σ_j exp(x_j))
 178 
 179      the input to this function is essentially a two 
 180      dimensional matrix (4D array whose last two axes
 181      have only one element), which is simply a set of
 182      vectors.
 183 
 184  */
 185   __device__ __host__
 186   array2<maxB,nC>& logsoftmax(array4<maxB,nC,1,1>& x) {
 187     const idx_t B = x.B;
 188     for (long b = 0; b < B; b++) {
 189       long m = 0;
 190       for (long c = 0; c < nC; c++) {
 191         m = (x(b,m,0,0) < x(b,c,0,0) ? c : m);
 192       }
 193       real s = 0.0;
 194       for (long c = 0; c < nC; c++) {
 195         lsm(b,c) = x(b,c,0,0) - x(b,m,0,0);
 196         s += exp(lsm(b,c));
 197       }
 198       for (long c = 0; c < nC; c++) {
 199         lsm(b,c) -= log(s);
 200       }
 201     }
 202     return lsm;
 203   }

 314   __device__ __host__
 315   void backward_base(vec<maxB>& gy) {
 316     const idx_t B = gy.n;
 317     gx.set_n_rows(B);
 318     ivec<maxB>& t = *t_ptr;
 319     for (idx_t b = 0; b < B; b++) {
 320       for (idx_t c = 0; c < nC; c++) {
 321         if (c == t(b)) {
 322           gx(b,c,0,0) = gy(b) * (-1 + exp(lsm(b,c)));
 323         } else {
 324           gx(b,c,0,0) = gy(b) * exp(lsm(b,c));
 325         }
 326       }
 327     }
 328   }

 405 /**
 406    @brief check the gradient computation of a softmaxcrossentropy layer
 407    @param (opt) command line option
 408    @param (lgr) logger 
 409    @param (rg) random number generator
 410    @param (B) the number of images
 411    @sa softmaxcrossentropy_main
 412    @details it first makes a layer object with initial weights W 
 413    and generates an input (x and t).
 414    it then creates two layers whose weights are slightly different
 415    from the original one by dw/2 (i.e., w-dw/2 and w+dw/2), as well as
 416    two inputs slighly different from the original inputs by dx/2
 417    (x-dx/2 and x+dx/2).  it then computes L(w,x), L(x-dw/2,x-dx/2) and
 418    L(w+dw/2,x+dw/2) and check if L(x+dw/2,x+dx/2)-L(x-dw/2,x-dx/2)
 419    is close to ∂L/∂x dx + ∂L/∂w dw.  ∂L/∂x and ∂L/∂w are obtained
 420    by backward computation. This is essentially checking if
 421    the gradients obtained by backward computation correctly approximates
 422    the diff of the output.
 423 */
 424 template<idx_t maxB,idx_t nC>
 425   static real softmaxcrossentropy_grad_check_rand(cmdline_opt opt, logger * lgr, rnd_gen_t& rg, idx_t B) {
     /*   */
 426   /* initialize softmax */
 427   SoftmaxCrossEntropy<maxB,nC> * smxe = new SoftmaxCrossEntropy<maxB,nC>();
 428   smxe->init(opt, lgr);
 429   smxe->make_dev();
 430   smxe->to_dev();
 431   /* make copies */
 432   SoftmaxCrossEntropy<maxB,nC>* smxe_minus = smxe->copy();
 433   SoftmaxCrossEntropy<maxB,nC>* smxe_plus  = smxe->copy();
 434   /* make coefficients to make the single loss value */
 435   vec<maxB> * alpha = new vec<maxB>();
 436   alpha->make_dev(opt.gpu_algo);
 437   alpha->init_uniform(B, rg, 0.0, 1.0);
 438   alpha->to_dev();
 439   /* make input (x) */
 440   array4<maxB,nC,1,1> * x = new array4<maxB,nC,1,1>();
 441   x->make_dev(opt.gpu_algo);
 442   x->init_uniform(B, rg, 0.0, 1.0);
 443   x->to_dev();
 444   /* make input (t) */
 445   ivec<maxB> * t = new ivec<maxB>();
 446   t->make_dev(opt.gpu_algo);
 447   t->init_uniform(B, rg, 0, nC);
 448   t->to_dev();
 449   /* forward and backward */
 450   vec<maxB>& y = smxe->forward(*x, *t);
 451   array4<maxB,nC,1,1>& gx = smxe->backward(*alpha);
 452   smxe->to_host();
 453 
 454   /* make dx */
 455   real e = 1.0e-4;
 456   array4<maxB,nC,1,1> * dx = new array4<maxB,nC,1,1>();
 457   dx->init_uniform(B, rg, -e, e);
 458   /* make x - dx/2 and x + dx/2 */
 459   array4<maxB,nC,1,1> * x_minus = new array4<maxB,nC,1,1>(*x);
 460   x_minus->make_dev(opt.gpu_algo);
 461   array4<maxB,nC,1,1> * x_plus  = new array4<maxB,nC,1,1>(*x);
 462   x_plus->make_dev(opt.gpu_algo);
 463   x_minus->update(-0.5, *dx);
 464   x_plus->update( 0.5, *dx);
 465   x_minus->to_dev();
 466   x_plus->to_dev();
 467     
 468   /* send copies to gpu */
 469   smxe_minus->to_dev();
 470   smxe_plus->to_dev();
 471   /* make y(x-dx/2), y(x+dx/2) */
 472   vec<maxB>& y_minus = smxe_minus->forward(*x_minus, *t);
 473   vec<maxB>& y_plus  = smxe_plus->forward(*x_plus, *t);
 474   /* get the result back to host */
 475   y_minus.to_host();
 476   y_plus.to_host();
 477 
 478   /* get the single loss values */
 479   real L_minus = alpha->dot(y_minus);
 480   real L       = alpha->dot(y);
 481   real L_plus  = alpha->dot(y_plus);
 482   /* various inner products */
 483   real gx_gx = gx.dot(gx);                         /* ∂L/∂x・∂L/∂x */
 484   real dx_dx = dx->dot(*dx);                       /* ∂L/∂x・dx */
 485   real gx_dx = gx.dot(*dx);                        /* dx・dx */
 486   real gw_gw = 0;                                  /* ∂L/∂w・∂L/∂w */
 487   real dw_dw = 0;                                  /* ∂L/∂w・dw */
 488   real gw_dw = 0;                                  /* dw・dw */
 489 
 490   real rel_e = show_error(gx_gx, dx_dx, gx_dx, gw_gw, dw_dw, gw_dw, L_minus, L, L_plus);
 491   /* clean up */
 492   smxe->del_dev();
 493   smxe_minus->del_dev();
 494   smxe_plus->del_dev();
 495   alpha->del_dev();
 496   x->del_dev();
 497   t->del_dev();
 498   dx->del_dev();
 499   x_minus->del_dev();
 500   x_plus->del_dev();
 501 
 502   delete smxe;
 503   delete smxe_minus;
 504   delete smxe_plus;
 505   delete alpha;
 506   delete x;
 507   delete t;
 508   delete dx;
 509   delete x_minus;
 510   delete x_plus;
 511   return rel_e;
 512 }

 514 /**
 515    @brief entry point of this header file
 516    @param (argc) the number of command line args
 517    @param (argv) command line args
 518    @sa softmaxcrossentropy_grad_check_rand
 519    @details if this header file is included from
 520    a main C++ file and define softmaxcrossentropy_main to be main
 521    (e.g., with -Dsoftmaxcrossentropy_main=main), then this
 522    function becomes th main function of the executable.
 523    it calls softmaxcrossentropy_grad_check_rand repeatedly to test
 524    the implementation of VGG network.
 525 */
 526 int softmaxcrossentropy_main(int argc, char ** argv) {
     /*   */
 527   cmdline_opt opt = parse_args(argc, argv);
 528   const idx_t maxB = MAX_BATCH_SIZE;
 529   const idx_t B = min_i(maxB, opt.batch_sz);
 530   const idx_t nC = 10;
 531   const int n_checks = opt.iters;
 532   /* logger */
 533   logger lgr;
 534   lgr.start_log(opt);
 535   /* initialize random number generator */
 536   rnd_gen_t rg;
 537   rg.seed(opt.weight_seed);
 538   /* check errors */
 539   real max_e = 0.0;
 540   real sum_e = 0.0;
 541   for (int iter = 0; iter < n_checks; iter++) {
 542     printf("==== %d ====\n", iter);
 543     real e = softmaxcrossentropy_grad_check_rand<maxB,nC>(opt, &lgr, rg, B);
 544     max_e = max_r(max_e, e);
 545     sum_e += e;
 546   }
 547   printf("max relative error = %.9f\n", max_e);
 548   printf("avg relative error = %.9f\n", sum_e / n_checks);
 549   lgr.end_log();
 550   return 0;
 551 }
