 500 /**
 501    @brief check the gradient computation of a linear layer
 502    @param (opt) command line option
 503    @param (lgr) logger 
 504    @param (rg) random number generator
 505    @param (B) the number of images
 506    @sa linear_main
 507    @details it first makes a layer object with initial weights W 
 508    and generates an input (x and t).
 509    it then creates two layers whose weights are slightly different
 510    from the original one by dw/2 (i.e., w-dw/2 and w+dw/2), as well as
 511    two inputs slighly different from the original inputs by dx/2
 512    (x-dx/2 and x+dx/2).  it then computes L(w,x), L(x-dw/2,x-dx/2) and
 513    L(w+dw/2,x+dw/2) and check if L(x+dw/2,x+dx/2)-L(x-dw/2,x-dx/2)
 514    is close to L/x dx + L/w dw.  L/x and L/w are obtained
 515    by backward computation. This is essentially checking if
 516    the gradients obtained by backward computation correctly approximates
 517    the diff of the output.
 518 */
 519 template<idx_t maxB,idx_t IC,idx_t nC>
 520   real linear_grad_check_rand(cmdline_opt opt, logger * lgr, rnd_gen_t& rg, idx_t B) {
     /*   */
 521   /* initialize linear parameters */
 522   Linear<maxB,IC,nC> * linear = new Linear<maxB,IC,nC>();
 523   linear->init(opt, lgr, rg);
 524   linear->make_dev();
 525   linear->to_dev();
 526   /* make w - dw/2 and w + dw/2 */
 527   Linear<maxB,IC,nC> * linear_minus = linear->copy();
 528   Linear<maxB,IC,nC> * linear_plus  = linear->copy();
 529   /* make coefficients to make the single loss value */
 530   array4<maxB,nC,1,1> * alpha = new array4<maxB,nC,1,1>();
 531   alpha->make_dev(opt.gpu_algo);
 532   alpha->init_uniform(B, rg, 0.0, 1.0);
 533   alpha->to_dev();
 534   /* make input (x) */
 535   array4<maxB,IC,1,1> * x = new array4<maxB,IC,1,1>();
 536   x->make_dev(opt.gpu_algo);
 537   x->init_uniform(B, rg, 0.0, 1.0);
 538   x->to_dev();
 539   /* forward and backward */
 540   array4<maxB,nC,1,1>& y = linear->forward(*x);
 541   array4<maxB,IC,1,1>& gx = linear->backward(*alpha);
 542   /* ensure the gradient is back to host */
 543   linear->to_host();
 544 
 545   /* make dx */
 546   real e = 1.0e-4;
 547   array4<maxB,IC,1,1> * dx = new array4<maxB,IC,1,1>();
 548   dx->init_uniform(B, rg, -e, e);
 549   /* make x - dx/2 and x + dx/2 */
 550   array4<maxB,IC,1,1> * x_minus = new array4<maxB,IC,1,1>(*x);
 551   x_minus->make_dev(opt.gpu_algo);
 552   array4<maxB,IC,1,1> * x_plus  = new array4<maxB,IC,1,1>(*x);
 553   x_plus->make_dev(opt.gpu_algo);
 554   /* update on the host and send the to gpu */
 555   x_minus->update(-0.5, *dx);
 556   x_plus->update( 0.5, *dx);
 557   x_minus->to_dev();
 558   x_plus->to_dev();
 559     
 560   /* set gw to a random vector */
 561   linear_minus->rand_grad(rg, -e, e);
 562   linear_plus->set_grad(*linear_minus);
 563   /* send them to gpu */
 564   linear_minus->to_dev();
 565   linear_plus->to_dev();
 566   /* update weights using gw (update runs on gpu) */
 567   linear_minus->update(-0.5);      /* w -= dw/2 */
 568   linear_plus->update(0.5);        /* w += dw/2 */
 569   /* make y(w-dw/2,x-dx/2), y(w+dw/2,x+dx/2) */
 570   array4<maxB,nC,1,1>& y_minus = linear_minus->forward(*x_minus);
 571   array4<maxB,nC,1,1>& y_plus  = linear_plus->forward(*x_plus);
 572   /* get the result back to host */
 573   y_minus.to_host();
 574   y_plus.to_host();
 575 
 576   /* get the single loss values */
 577   real L_minus = alpha->dot(y_minus);
 578   real L       = alpha->dot(y);
 579   real L_plus  = alpha->dot(y_plus);
 580   /* various inner products */
 581   real gx_gx = gx.dot(gx);                         /* L/xL/x */
 582   real dx_dx = dx->dot(*dx);                       /* L/xdx */
 583   real gx_dx = gx.dot(*dx);                        /* dxdx */
 584   real gw_gw = linear->gw_dot_gw(*linear);             /* L/wL/w */
 585   real dw_dw = linear_minus->gw_dot_gw(*linear_minus); /* L/wdw */
 586   real gw_dw = linear->gw_dot_gw(*linear_minus);       /* dwdw */
 587 
 588   real rel_e = show_error(gx_gx, dx_dx, gx_dx, gw_gw, dw_dw, gw_dw, L_minus, L, L_plus);
 589   /* clean up */
 590   linear->del_dev();
 591   linear_minus->del_dev();
 592   linear_plus->del_dev();
 593   alpha->del_dev();
 594   x->del_dev();
 595   dx->del_dev();
 596   x_minus->del_dev();
 597   x_plus->del_dev();
 598 
 599   delete linear;
 600   delete linear_minus;
 601   delete linear_plus;
 602   delete alpha;
 603   delete x;
 604   delete dx;
 605   delete x_minus;
 606   delete x_plus;
 607   return rel_e;
 608 }

 624   const idx_t maxB = MAX_BATCH_SIZE;
 625   const idx_t B = min_i(maxB, opt.batch_sz);
 626   const idx_t IC = 512;

 629   /* logger */

 631   lgr.start_log(opt);

 635   /* check errors */
 636   real max_e = 0.0;
 637   real sum_e = 0.0;
