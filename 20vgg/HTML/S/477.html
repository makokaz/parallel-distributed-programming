 313   /**
 314      @brief calc the loss function of a mini-batch (x,t)
 315      @param (x) input images
 316      @param (t) true labels of images
 317      @sa backward
 318      @sa update
 319   */
 320   vec<maxB>& forward(array4<maxB,C0,H,W>& x, ivec<maxB>& t) {
 321     /* group 1 : (C0,H1,W1)->(C1,H2,W2) */
 322     array4<maxB,C1,H1,W1>&  x1 = block1_1.forward(x);
 323     array4<maxB,C1,H1,W1>&  x2 = dropout1_1.forward(x1);
 324     array4<maxB,C1,H1,W1>&  x3 = block1_2.forward(x2);
 325     array4<maxB,C1,H2,W2>&  x4 = max_pooling_2d1.forward(x3);
 326     /* group 2 : (C1,H2,W2)->(C2,H3,W3) */
 327     array4<maxB,C2,H2,W2>&  x5 = block2_1.forward(x4);
 328     array4<maxB,C2,H2,W2>&  x6 = dropout2_1.forward(x5);
 329     array4<maxB,C2,H2,W2>&  x7 = block2_2.forward(x6);
 330     array4<maxB,C2,H3,W3>&  x8 = max_pooling_2d2.forward(x7);
 331     /* group 3 : (C2,H3,W3)->(C3,H4,W4) */
 332     array4<maxB,C3,H3,W3>&  x9 = block3_1.forward(x8);
 333     array4<maxB,C3,H3,W3>& x10 = dropout3_1.forward(x9);
 334     array4<maxB,C3,H3,W3>& x11 = block3_2.forward(x10);
 335     array4<maxB,C3,H3,W3>& x12 = dropout3_2.forward(x11);
 336     array4<maxB,C3,H3,W3>& x13 = block3_3.forward(x12);
 337     array4<maxB,C3,H4,W4>& x14 = max_pooling_2d3.forward(x13);
 338     /* group 4 : (C3,H4,W4)->(C4,H5,W5) */
 339     array4<maxB,C4,H4,W4>& x15 = block4_1.forward(x14);
 340     array4<maxB,C4,H4,W4>& x16 = dropout4_1.forward(x15);
 341     array4<maxB,C4,H4,W4>& x17 = block4_2.forward(x16);
 342     array4<maxB,C4,H4,W4>& x18 = dropout4_2.forward(x17);
 343     array4<maxB,C4,H4,W4>& x19 = block4_3.forward(x18);
 344     array4<maxB,C4,H5,W5>& x20 = max_pooling_2d4.forward(x19);
 345     /* group 5 : (C4,H5,W5)->(C4,H6,W6) */
 346     array4<maxB,C4,H5,W5>& x21 = block5_1.forward(x20);
 347     array4<maxB,C4,H5,W5>& x22 = dropout5_1.forward(x21);
 348     array4<maxB,C4,H5,W5>& x23 = block5_2.forward(x22);
 349     array4<maxB,C4,H5,W5>& x24 = dropout5_2.forward(x23);
 350     array4<maxB,C4,H5,W5>& x25 = block5_3.forward(x24);
 351     array4<maxB,C4,H6,W6>& x26 = max_pooling_2d5.forward(x25);
 352     /* group 6 : (C4,H6,W6) -> classification -> loss */
 353     array4<maxB,C4,H6,W6>& x27 = dropout6_1.forward(x26);
 354     array4<maxB,C4,H6,W6>& x28 = fc1.forward(x27);
 355     array4<maxB,C4,H6,W6>& x29 = bn_fc1.forward(x28);
 356     array4<maxB,C4,H6,W6>& x30 = relu.forward(x29);
 357     array4<maxB,C4,H6,W6>& x31 = dropout6_2.forward(x30);
 358     array4<maxB,nC,H6,W6>& x32 = fc2.forward(x31);
 359     vec<maxB>& y = softmax_cross_entropy.forward(x32, t);
 360     return y;
 361   }
 362   /**
 363      @brief calc the gradient of loss wrt the input (x)
 364      @param (gy) gradient of loss with respect to the output
 365      @details calc the gradient of loss wrt the input. along the way,
 366      it also calculates the gradient of loss wrt weights for
 367      all sublayers that have weights. since this is the entire
 368      network, gy is actually a vector whose components are all 1.
 369      (loss = sum of losses of each data).
 370      @sa forward
 371      @sa update
 372   */
 373   array4<maxB,C0,H1,W1>& backward(vec<maxB>& gy) {
 374     /* group 6 */
 375     array4<maxB,nC,H6,W6>& g32 = softmax_cross_entropy.backward(gy);
 376     array4<maxB,C4,H6,W6>& g31 = fc2.backward(g32);
 377     array4<maxB,C4,H6,W6>& g30 = dropout6_2.backward(g31);
 378     array4<maxB,C4,H6,W6>& g29 = relu.backward(g30);
 379     array4<maxB,C4,H6,W6>& g28 = bn_fc1.backward(g29);
 380     array4<maxB,C4,H6,W6>& g27 = fc1.backward(g28);
 381     array4<maxB,C4,H6,W6>& g26 = dropout6_1.backward(g27);
 382     /* group 5 */
 383     array4<maxB,C4,H5,W5>& g25 = max_pooling_2d5.backward(g26);
 384     array4<maxB,C4,H5,W5>& g24 = block5_3.backward(g25);
 385     array4<maxB,C4,H5,W5>& g23 = dropout5_2.backward(g24);
 386     array4<maxB,C4,H5,W5>& g22 = block5_2.backward(g23);
 387     array4<maxB,C4,H5,W5>& g21 = dropout5_1.backward(g22);
 388     /* group 4 */
 389     array4<maxB,C4,H5,W5>& g20 = block5_1.backward(g21);
 390     array4<maxB,C4,H4,W4>& g19 = max_pooling_2d4.backward(g20);
 391     array4<maxB,C4,H4,W4>& g18 = block4_3.backward(g19);
 392     array4<maxB,C4,H4,W4>& g17 = dropout4_2.backward(g18);
 393     array4<maxB,C4,H4,W4>& g16 = block4_2.backward(g17);
 394     array4<maxB,C4,H4,W4>& g15 = dropout4_1.backward(g16);
 395     /* group 3 */
 396     array4<maxB,C3,H4,W4>& g14 = block4_1.backward(g15);
 397     array4<maxB,C3,H3,W3>& g13 = max_pooling_2d3.backward(g14);
 398     array4<maxB,C3,H3,W3>& g12 = block3_3.backward(g13);
 399     array4<maxB,C3,H3,W3>& g11 = dropout3_2.backward(g12);
 400     array4<maxB,C3,H3,W3>& g10 = block3_2.backward(g11);
 401     array4<maxB,C3,H3,W3>&  g9 = dropout3_1.backward(g10);
 402     /* group 2 */
 403     array4<maxB,C2,H3,W3>&  g8 = block3_1.backward(g9);
 404     array4<maxB,C2,H2,W2>&  g7 = max_pooling_2d2.backward(g8);
 405     array4<maxB,C2,H2,W2>&  g6 = block2_2.backward(g7);
 406     array4<maxB,C2,H2,W2>&  g5 = dropout2_1.backward(g6);
 407     /* group 1 */
 408     array4<maxB,C1,H2,W2>&  g4 = block2_1.backward(g5);
 409     array4<maxB,C1,H1,W1>&  g3 = max_pooling_2d1.backward(g4);
 410     array4<maxB,C1,H1,W1>&  g2 = block1_2.backward(g3);
 411     array4<maxB,C1,H1,W1>&  g1 = dropout1_1.backward(g2);
 412     array4<maxB,C0,H1,W1>&  g0 = block1_1.backward(g1);
 413     return g0;
 414   }
 415   int log_minibatch(idx_t start_offset) {
 416     array2<maxB,nC>& lsm = softmax_cross_entropy.lsm;
 417     lsm.to_host();
 418     const idx_t B = idxs.n;
 419     int correct = 0;
 420     for (idx_t b = 0; b < B; b++) {
 421       /* get the prediction from logsoftmax */
 422       idx_t pred_class = 0;
 423       for (idx_t c = 0; c < nC; c++) {
 424         if (lsm(b,pred_class) < lsm(b,c)) {
 425           pred_class = c;
 426         }
 427       }
 428       if (pred_class == t(b)) {
 429         correct++;
 430       }
 431       lgr->log(1, "sample %d image %d pred %d truth %d",
 432                start_offset + b, idxs(b), pred_class, t(b));
 433     }
 434     return correct;
 435   }
 436 
 437   /**
 438      @brief perform an entire iteration (= forward; backward; update)
 439      @param (x) input images (a mini batch)
 440      @param (t) true labels
 441      @param (eta) learning rate
 442      @sa forward
 443      @sa backward
 444      @sa update
 445      @details do everything on a mini-batch. forward calculates the 
 446      loss wrt x and t; backward calculates the gradient 
 447      of loss wrt x and weights; update updates weights with the
 448      gradients and the learning rate.
 449   */
 450   real forward_backward_update(array4<maxB,C0,H,W>& x, ivec<maxB>& t, real eta) {
 451     const idx_t B = x.B;
 452     /* forward */
 453     vec<maxB>& y = forward(x, t);
 454     /* a vector (1,1,1,...) to make the single loss value from loss of each sample */
 455     gy.init_const(B, 1.0);
 456     gy.to_dev();
 457     /* backward (set weights of all sublayers) */
 458     backward(gy);
 459     /* update */
 460     real e = eta / B;
 461     update(-e);
 462     /* get the loss of each sample back to host if we are working on GPU */
 463     y.to_host();
 464     real L = gy.dot(y);
 465     return L;
 466   }
 467   /* member functions below assume data are on the host.
 468      they are only for checking (debugging) implementations */
 469   /**

 498   /**

 500      @param (o) the object from which gradients get copied
 501      @details transfer gradients of o to this object

 503   void set_grad(VGG<maxB,C0,H,W,K,S,C1,nC>& o) {
 504     block1_1.set_grad(o.block1_1);
 505     block1_2.set_grad(o.block1_2);

 507     block2_1.set_grad(o.block2_1);

 526   /**
 527      @brief take the inner product of gradients
 528      @param (b) the object to take the inner product with
 529      @details take the inner product of this object's 
 530      gradients and b's gradients
 531   */
 532   real gw_dot_gw(VGG<maxB,C0,H,W,K,S,C1,nC>& b) {
 533     VGG<maxB,C0,H,W,K,S,C1,nC>& a = *this;
 534     real s = 0.0;
 535     s += a.block1_1.gw_dot_gw(b.block1_1);
 536     s += a.block1_2.gw_dot_gw(b.block1_2);
 537     s += a.block2_1.gw_dot_gw(b.block2_1);
 538     s += a.block2_2.gw_dot_gw(b.block2_2);
 539     s += a.block3_1.gw_dot_gw(b.block3_1);
 540     s += a.block3_2.gw_dot_gw(b.block3_2);
 541     s += a.block3_3.gw_dot_gw(b.block3_3);
 542     s += a.block4_1.gw_dot_gw(b.block4_1);
 543     s += a.block4_2.gw_dot_gw(b.block4_2);
 544     s += a.block4_3.gw_dot_gw(b.block4_3);
 545     s += a.block5_1.gw_dot_gw(b.block5_1);
 546     s += a.block5_2.gw_dot_gw(b.block5_2);
 547     s += a.block5_3.gw_dot_gw(b.block5_3);
 548     s += a.fc1.gw.dot(b.fc1.gw);
 549     s += a.bn_fc1.gw_dot_gw(b.bn_fc1);
 550     s += a.fc2.gw.dot(b.fc2.gw);
 551     return s;
 552   }

 555 /**
 556    @brief check the gradient computation of a VGG network
 557    @param (opt) command line option
 558    @param (lgr) logger 
 559    @param (rg) random number generator
 560    @param (B) the number of images
 561    @sa vgg_main
 562    @details it first makes a VGG network with initial weights W 
 563    and generates an input (x and t).
 564    it then creates two VGG networks whose weights are slightly different
 565    from the original one by dw/2 (i.e., w-dw/2 and w+dw/2), as well as
 566    two inputs slighly different from the original inputs by dx/2
 567    (x-dx/2 and x+dx/2).  it then computes L(w,x), L(x-dw/2,x-dx/2) and
 568    L(w+dw/2,x+dw/2) and check if L(x+dw/2,x+dx/2)-L(x-dw/2,x-dx/2)
 569    is close to L/x dx + L/w dw.  L/x and L/w are obtained
 570    by backward computation. This is essentially checking if
 571    the gradients obtained by backward computation correctly approximates
 572    the diff of the output.
 573 */
 574 template<idx_t maxB,idx_t C0,idx_t H,idx_t W,idx_t K,idx_t S,idx_t C1,idx_t nC>
 575   static real vgg_grad_check_rand(cmdline_opt opt, logger * lgr, rnd_gen_t& rg, idx_t B) {
     /*   */
 576   /* initialize vgg parameters */
 577   VGG<maxB,C0,H,W,K,S,C1,nC> * vgg = new VGG<maxB,C0,H,W,K,S,C1,nC>();
 578   vgg->init(opt, lgr, rg);
 579   vgg->make_dev();
 580   vgg->to_dev();
 581   /* make w - dw/2 and w + dw/2 */
 582   VGG<maxB,C0,H,W,K,S,C1,nC> * vgg_minus = vgg->copy();
 583   VGG<maxB,C0,H,W,K,S,C1,nC> * vgg_plus  = vgg->copy();
 584   /* make coefficients to make the single loss value */
 585   vec<maxB> * alpha = new vec<maxB>();
 586   alpha->make_dev(opt.gpu_algo);
 587   alpha->init_uniform(B, rg, 0.0, 1.0);
 588   alpha->to_dev();
 589   /* make input (x) */
 590   array4<maxB,C0,H,W> * x = new array4<maxB,C0,H,W>();
 591   x->make_dev(opt.gpu_algo);
 592   x->init_uniform(B, rg, 0.0, 1.0);
 593   x->to_dev();
 594   /* make input (t) */
 595   ivec<maxB>* t = new ivec<maxB>();
 596   t->make_dev(opt.gpu_algo);
 597   t->init_uniform(B, rg, 0, nC);
 598   t->to_dev();
 599   /* forward and backward */
 600   vec<maxB>& y = vgg->forward(*x, *t);
 601   array4<maxB,C0,H,W>& gx = vgg->backward(*alpha);
 602   /* ensure the gradient is back to host */
 603   vgg->to_host();
 604     
 605   /* make dx */
 606   real e = 1.0e-4;
 607   array4<maxB,C0,H,W> * dx = new array4<maxB,C0,H,W>();
 608   dx->init_uniform(B, rg, -e, e);
 609   /* make x - dx/2 and x + dx/2 */
 610   array4<maxB,C0,H,W> * x_minus = new array4<maxB,C0,H,W>(*x);
 611   x_minus->make_dev(opt.gpu_algo);
 612   array4<maxB,C0,H,W> * x_plus  = new array4<maxB,C0,H,W>(*x);
 613   x_plus->make_dev(opt.gpu_algo);
 614   /* update on the host and send the to gpu */
 615   x_minus->update(-0.5, *dx);
 616   x_plus->update( 0.5, *dx);
 617   x_minus->to_dev();
 618   x_plus->to_dev();
 619     
 620   /* set gw to a random vector */
 621   vgg_minus->rand_grad(rg, -e, e);
 622   vgg_plus->set_grad(*vgg_minus);
 623   /* send them to gpu */
 624   vgg_minus->to_dev();
 625   vgg_plus->to_dev();
 626   /* update weights using gw (update runs on gpu) */
 627   vgg_minus->update(-0.5);      /* w -= dw/2 */
 628   vgg_plus->update(0.5);        /* w += dw/2 */
 629   /* make y(w-dw/2,x-dx/2), y(w+dw/2,x+dx/2) */
 630   vec<maxB>& y_minus = vgg_minus->forward(*x_minus, *t);
 631   vec<maxB>& y_plus  = vgg_plus->forward(*x_plus, *t);
 632   /* get the result back to host */
 633   y_minus.to_host();
 634   y_plus.to_host();
 635 
 636   /* get the single loss values */
 637   real L_minus = alpha->dot(y_minus);
 638   real L       = alpha->dot(y);
 639   real L_plus  = alpha->dot(y_plus);
 640   /* various inner products */
 641   real gx_gx = gx.dot(gx);                         /* L/xL/x */
 642   real dx_dx = dx->dot(*dx);                       /* L/xdx */
 643   real gx_dx = gx.dot(*dx);                        /* dxdx */
 644   real gw_gw = vgg->gw_dot_gw(*vgg);             /* L/wL/w */
 645   real dw_dw = vgg_minus->gw_dot_gw(*vgg_minus); /* L/wdw */
 646   real gw_dw = vgg->gw_dot_gw(*vgg_minus);       /* dwdw */
 647   
 648   real rel_e = show_error(gx_gx, dx_dx, gx_dx, gw_gw, dw_dw, gw_dw, L_minus, L, L_plus);
 649   /* clean up */
 650   vgg->del_dev();
 651   vgg_minus->del_dev();
 652   vgg_plus->del_dev();
 653   alpha->del_dev();
 654   x->del_dev();
 655   dx->del_dev();
 656   x_minus->del_dev();
 657   x_plus->del_dev();
 658 
 659   delete vgg;
 660   delete vgg_minus;
 661   delete vgg_plus;
 662   delete alpha;
 663   delete x;
 664   delete dx;
 665   delete x_minus;
 666   delete x_plus;
 667   return rel_e;
 668 }

 670 /**
 671    @brief entry point of this header file
 672    @param (argc) the number of command line args
 673    @param (argv) command line args
 674    @sa vgg_grad_check_rand
 675    @details if this header file is included from
 676    a main C++ file and define vgg_main to be main
 677    (e.g., with -Dvgg_main=main), then this
 678    function becomes th main function of the executable.
 679    it calls vgg_grad_check_rand repeatedly to test
 680    the implementation of VGG network.
 681 */
 682 int vgg_main(int argc, char ** argv) {
     /*   */
 683   cmdline_opt opt = parse_args(argc, argv);
 684   const idx_t maxB = MAX_BATCH_SIZE;
 685   const idx_t B = min_i(maxB, opt.batch_sz);
 686   const int C0 = 3;
 687   const int H = 32;
 688   const int W = 32;
 689   const int K = 1;
 690   const int S = 2;
 691   const int C1 = 64;
 692   const int nC = 10;
 693   const int n_checks = opt.iters;
 694   /* logger */
 695   logger lgr;
 696   lgr.start_log(opt);
 697   /* initialize random number generator */
 698   rnd_gen_t rg;
 699   rg.seed(opt.sample_seed);
 700   /* check errors */
 701   real max_e = 0.0;
 702   real sum_e = 0.0;
 703   for (int iter = 0; iter < n_checks; iter++) {
 704     printf("==== %d ====\n", iter);
 705     real e = vgg_grad_check_rand<maxB,C0,H,W,K,S,C1,nC>(opt, &lgr, rg, B);
 706     max_e = max_r(max_e, e);
 707     sum_e += e;
 708   }
 709   printf("max relative error = %.9f\n", max_e);
 710   printf("avg relative error = %.9f\n", sum_e / n_checks);
 711   lgr.end_log();
 712   return 0;
 713 }
