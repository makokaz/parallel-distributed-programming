 269    @brief parse command line args and make a command line object

 271 static cmdline_opt parse_args(int argc, char ** argv) {
     /*   */
 272   char * prog = argv[0];
 273   cmdline_opt opt;
 274   while (1) {
 275     int option_index = 0;
 276     int c = getopt_long(argc, argv,
 277                         "a:b:d:D:l:m:v:h", long_options, &option_index);
 278     if (c == -1) break;
 279     switch (c) {
 280     case 0:
 281       {
 282         const char * o = long_options[option_index].name;
 283         if (strcmp(o, "partial_data") == 0) {
 284           opt.partial_data = atol(optarg);
 285         } else if (strcmp(o, "single_batch") == 0) {
 286           opt.single_batch = atoi(optarg);
 287         } else if (strcmp(o, "dropout") == 0) {
 288           opt.dropout = (atoi(optarg) != 0);
 289         } else if (strcmp(o, "validate_ratio") == 0) {
 290           opt.validate_ratio = atof(optarg);
 291         } else if (strcmp(o, "validate_interval") == 0) {
 292           opt.validate_interval = atof(optarg);
 293         } else if (strcmp(o, "sample_seed") == 0) {
 294           opt.sample_seed = atol(optarg);
 295         } else if (strcmp(o, "weight_seed") == 0) {
 296           opt.weight_seed = atol(optarg);
 297         } else if (strcmp(o, "dropout_seed") == 0) {
 298           opt.dropout_seed = atol(optarg);
 299         } else if (strcmp(o, "partial_data_seed") == 0) {
 300           opt.partial_data_seed = atol(optarg);
 301         } else if (strcmp(o, "grad_dbg") == 0) {
 302           opt.grad_dbg = atoi(optarg);
 303         } else if (strcmp(o, "log") == 0) {
 304           opt.log = strdup(optarg);
 305         } else {
 306           fprintf(stderr,
 307                   "bug:%s:%d: should handle option %s\n",
 308                   __FILE__, __LINE__, o);
 309           opt.error = 1;
 310           return opt;
 311         }
 312       }
 313       break;
 314     case 'v':
 315       opt.verbose = atoi(optarg);
 316       break;
 317     case 'd':
 318       opt.cifar_data = strdup(optarg);
 319       break;
 320     case 'D':
 321       opt.cifar_data_dump = strdup(optarg);
 322       break;
 323     case 'b':
 324       opt.batch_sz = atoi(optarg);
 325       break;
 326     case 'a':
 327       opt.algo_s = strdup(optarg);
 328       break;
 329     case 'l':
 330       opt.learnrate = atof(optarg);
 331       break;
 332     case 'm':
 333       opt.iters = atol(optarg);
 334       break;
 335     case 'h':
 336       opt.help = 1;
 337       break;
 338     default: /* '?' */
 339       usage(prog);
 340       opt.error = 1;
 341       return opt;
 342     }
 343   }
 344   if (opt.batch_sz > MAX_BATCH_SIZE) {
 345     fprintf(stderr, "error: cannot specify --batch_sz (%d) > MAX_BATCH_SIZE (%d)\n",
 346             opt.batch_sz, MAX_BATCH_SIZE);
 347     opt.error = 1;
 348     return opt;
 349   }
 350   opt.algo = parse_algo(opt.algo_s);
 351   if (opt.algo == algo_invalid) {
 352     fprintf(stderr, "error: invalid algorithm (%s)\n", opt.algo_s);
 353     opt.error = 1;
 354     return opt;
 355   }
 356   opt.gpu_algo = algo_is_gpu(opt.algo_s, opt.algo);
 357 #if !__NVCC__
 358   if (opt.gpu_algo) {
 359     fprintf(stderr, "error: --gpu 1 allowed only with nvcc\n");
 360     opt.error = 1;
 361     return opt;
 362   }
 363 #endif
 364   return opt;
 365 }

 409 static tsc_t get_tsc() {
     /*   */
 410   struct timespec ts[1];
 411   tsc_t t;
 412   if (clock_gettime(CLOCK_REALTIME, ts) == -1) {
 413     perror("clock_gettime"); bail();
 414   }
 415   t.ns = ts->tv_sec * 1000000000L + ts->tv_nsec;
 416   return t;
 417 }

 419 /**
 420    @brief pseudo random number generator
 421    crafted from man erand48 + libc source
 422 */
 423 struct rnd_gen_t {
 424   uint64_t x;                   /**< random number state */
 425   /**
 426      @brief set next state
 427    */
 428   __device__ __host__
 429   void next() {
 430     const uint64_t __a = 0x5deece66dull;
 431     const uint64_t __c = 0xb;
 432     const uint64_t mask = (1UL << 48) - 1;
 433     x = (x * __a + __c) & mask;
 434   }
 435   /**
 436      @brief return a random number between 0 and 1
 437    */
 438   __device__ __host__
 439   double rand01() {
 440     union ieee754_double temp;
 441     /* Compute next state.  */
 442     next();
 443     /* Construct a positive double with the 48 random bits distributed over
 444        its fractional part so the resulting FP number is [0.0,1.0).  */
 445     temp.ieee.negative = 0;
 446     temp.ieee.exponent = IEEE754_DOUBLE_BIAS;
 447     temp.ieee.mantissa0 = (x >> 28) & ((1UL << 20) - 1); /* 20 bit */
 448     temp.ieee.mantissa1 = (x & ((1UL << 28) - 1)) << 4;
 449     /* Please note the lower 4 bits of mantissa1 are always 0.  */
 450     return temp.d - 1.0;
 451   }
 452   /**
 453      @brief return a long between 0 to 2^31 - 1
 454    */
 455   __device__ __host__
 456   long randi32() {
 457     /* Compute next state.  */
 458     next();
 459     /* Store the result.  */
 460     return (x >> 17) & ((1UL << 31) - 1);
 461   }
 462   /**
 463      @brief return a real between a and b
 464    */
 465   __device__ __host__
 466   double rand(double a, double b) {
 467     return a + (b - a) * rand01();
 468   }
 469   /**
 470      @brief return a long between a and b
 471    */
 472   __device__ __host__
 473   long randi(long a, long b) {
 474     return a + randi32() % (b - a);
 475   }
 476   /**
 477      @brief generate a random number from a normal distribution
 478      of mean=mu and standard deviation=sigma.
 479      @details see
 480      https://en.wikipedia.org/wiki/Normal_distribution
 481      for how the following method works 
 482   */
 483   __device__ __host__
 484   real rand_normal(real mu, real sigma) {
 485     real u = rand01();
 486     real v = rand01();
 487     real x = sqrt(-2.0 * log(u)) * cos(2.0 * M_PI * v);
 488     return mu + x * sigma;
 489   }
 490   /**
 491      @brief return the current state of the generator
 492   */
 493   __device__ __host__
 494   long get_state() {
 495     return x;
 496   }
 497   /**
 498      @brief set state of the generator
 499   */
 500   __device__ __host__
 501   void seed(uint64_t y) {
 502     x = y;
 503   }
 504 };

 506 /**
 507    @brief show various errors 
 508    @param (gx_gx) L/xL/x
 509    @param (dx_dx) dxdx
 510    @param (gx_dx) L/xdx
 511    @param (gw_gw) L/wL/w
 512    @param (dw_dw) dwdw
 513    @param (gw_dw) L/wdw
 514    @param (L_minus) L(w-dw,x-dx)
 515    @param (L) L(w,x)
 516    @param (L_plus) L(w+dw,x+dx)
 517  */
 518 static real show_error(real gx_gx, real dx_dx, real gx_dx,
     /*   */
 519                        real gw_gw, real dw_dw, real gw_dw,
 520                        real L_minus, real L, real L_plus) {
 521   printf("|L/x|   = %.9f\n", sqrt(gx_gx));
 522   printf("|dx|      = %.9f\n", sqrt(dx_dx));
 523   printf("L/xdx = %.9f\n", gx_dx);
 524   printf("|L/w|   = %.9f\n", sqrt(gw_gw));
 525   printf("|dw|      = %.9f\n", sqrt(dw_dw));
 526   printf("L/wdw = %.9f\n", gw_dw);
 527   printf("L- = %.9f\n", L_minus);
 528   printf("L  = %.9f\n", L);
 529   printf("L+ = %.9f\n", L_plus);
 530   real dL = L_plus - L_minus;
 531   real A = gx_dx + gw_dw;
 532   real B = dL;
 533   real e = (A == B ? 0.0 : fabs(A - B) / max_r(fabs(A), fabs(B)));
 534   printf("A = L/xdx + L/wdw = %.9f\n", gx_dx + gw_dw);
 535   printf("B = L = %.9f\n", dL);
 536   printf("relative error = |A-B|/max(|A|,|B|) = %.9f\n", e);
 537   return e;
 538 }

 550   char * cur_time_str() {
 551     time_t t = time(NULL);
 552     char * time_s = ctime(&t);
 553     if (!time_s) {
 554       perror("ctime_r");
 555       exit(EXIT_FAILURE);
 556     }
 557     int len = strlen(time_s);
 558     time_s[len-1] = 0;
 559     return time_s;

 561   /**
 562      @brief write a formatted string to the log and may be to standard out
 563      @param (level) the level of this entry. if opt.verbose>=level, then 
 564      the string will be output to the standard out (in addition to the log file)
 565      @param (format) the printf-like format string
 566    */
 567   int log(int level, const char * format, ...) {
 568     tsc_t t = get_tsc();
 569     long dt = t.ns - t0.ns;
 570     if (log_fp) {
 571       va_list ap;
 572       fprintf(log_fp, "%ld: ", dt);
 573       va_start(ap, format);
 574       vfprintf(log_fp, format, ap);
 575       va_end(ap);
 576       fprintf(log_fp, "\n");
 577     }
 578     if (opt.verbose>=level) {
 579       va_list ap;
 580       fprintf(stdout, "%ld: ", dt);
 581       va_start(ap, format);
 582       vfprintf(stdout, format, ap);
 583       va_end(ap);
 584       fprintf(stdout, "\n");
 585       fflush(stdout);
 586     }
 587     return 1;
 588   }

 598     log(2, "open a log %s", cur_time_str());

 722 /**
 723    @brief entry point
 724  */
 725 int vgg_util_main(int argc, char ** argv) {
     /*   */
 726   cmdline_opt opt = parse_args(argc, argv);
 727   if (opt.error || opt.help) usage(argv[0]);
 728   rnd_gen_t rg;
 729   rg.seed(opt.sample_seed);
 730   bail();
 731   min_i(1,2);
 732   max_i(3,4);
 733   min_r(1.2,3.4);
 734   max_r(5.6,7.8);
 735   return 0;
 736 }
