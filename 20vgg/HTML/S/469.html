  53      @brief count the number of images in a file (should be 10000)
  54      @param (cifar_bin) the name of the file to read data from

  56   long get_n_data_in_file(const char * cifar_bin) {
  57     struct stat sb[1];
  58     if (lstat(cifar_bin, sb) == -1) {
  59       perror("lstat");
  60       fprintf(stderr,
  61               "error: could not find the cifar data file %s. Specify a right file name with -d FILENAME or do this so that it can find the default file (cifar-10-batches-bin/data_batch_1.bin).\n"
  62               "\n"
  63               "$ ln -s /home/tau/cifar10/cifar-10-batches-bin\n",
  64               cifar_bin);
  65       exit(1);
  66     }
  67     long sz = sb->st_size;
  68     long sz1 = IC * H * W + 1;
  69     assert(sz % sz1 == 0);
  70     return sz / sz1;
  71   }
  72   long get_decimal_digits(long n) {
  73     /* n <  10 -> 1 
  74        n < 100 -> 2
  75        find minimum d s.t. n < 10^d
  76 */
  77     long d = 1;
  78     long x = 10;                // x = 10^d
  79     for (d = 1; x < n; d++) {
  80       x = x * 10;
  81     }
  82     assert(n <= x);
  83     return d;
  84   }

  86   /**
  87      @brief dump dataset into files
  88      @param (dataset) dataset
  89      @param (n_data) size of dataset
  90      @param (prefix) prefix of files, like "img/img_" (-> img/img_xxxxx.ppm)
  91    */
  92   int dump_cifar_files(cifar10_data_item<IC,H,W>* dataset, long n_data,
  93                        long n_digits, 
  94                        const char * prefix) {
  95     /* chars required for data numbers */
  96     long nd = get_decimal_digits(n_data);
  97     if (n_digits < nd) {
  98       n_digits = nd;
  99     }
 100     long len = strlen(prefix) + n_digits + strlen(".ppm");
 101     char filename[len + 1];
 102     char fmt[100];
 103     /* -> "%s%05d.ppm" */
 104     int w = snprintf(fmt, 100, "%%s%%0%ldld.ppm", n_digits);
 105     assert(w <= 100);
 106     for (long d = 0; d < n_data; d++) {
 107       /* make a filename like "img/img_01234.ppm" */
 108       int w = snprintf(filename, len + 1, fmt, prefix, d);
 109       assert(w == len);
 110       FILE * wp = fopen(filename, "w");
 111       if (!wp) {
 112         perror("fopen");
 113         fprintf(stderr, "%s\n", filename);
 114         exit(1);
 115       }
 116       fprintf(wp, "P3 %d %d 255\n", W, H);
 117       for (idx_t i = 0; i < H; i++) {
 118         for (idx_t j = 0; j < W; j++) {
 119           for (idx_t c = 0; c < IC; c++) {
 120             fprintf(wp, " %d", dataset[d].rgb[c][i][j]);
 121           }
 122           fprintf(wp, "\n");
 123         }
 124       }
 125       fclose(wp);
 126     }
 127     return 1;                   /* OK */
 128   }

 131   /**
 132      @brief load training/validation data from the file 
 133      @param (lgr) logger
 134      @param (cifar_bin) the name of the file to read data from
 135      @param (n_samples) the number of data used 
 136      @param (sample_seed) seed of the random number generator to
 137      pick training and validation data
 138      @param (validate_ratio) leave this much for validation (<1.0)
 139    */
 140   int load(logger& lgr,
 141            const char * cifar_bin, long n_samples,
 142            long sample_seed, double validate_ratio,
 143            const char * dump_prefix) {
 144     long n_data_in_file = get_n_data_in_file(cifar_bin);
 145     if (n_samples == 0) {
 146       n_samples = n_data_in_file;
 147     } else if (n_samples > n_data_in_file) {
 148       lgr.log(1, "specified number of samples (%ld) exeeds data in file (%ld). truncated\n",
 149               n_samples, n_data_in_file);
 150       n_samples = n_data_in_file;
 151     }
 152     n_data = n_samples;
 153     n_train = max_i(1, n_data - n_data * validate_ratio);
 154     n_validate = n_data - n_train;
 155     lgr.log(1, "loading %ld/%ld training/validation data from %s starts",
 156             n_train, n_validate, cifar_bin);
 157     if (n_validate == 0) {
 158       lgr.log(1, "warning: no data left for validation (validation not performed)");
 159     }
 160     
 161     dataset = new cifar10_data_item<IC,H,W> [n_data_in_file];
 162     //validate = dataset;
 163     //train = dataset + n_validate;
 164     FILE * fp = fopen(cifar_bin, "rb");
 165     if (!fp) { perror("fopen"); exit(1); }
 166     for (int k = 0; k < n_data_in_file; k++) {
 167       unsigned char label[1];
 168       unsigned char rgb[IC][H][W];
 169       size_t r = fread(label, sizeof(label), 1, fp);
 170       if (ferror(fp)) { perror("fread"); exit(1); }
 171       if (feof(fp)) break;
 172       r = fread(rgb, sizeof(rgb), 1, fp);
 173       if (r != 1) { perror("fread"); exit(1); }
 174       int max_value = 0;
 175       for (idx_t c = 0; c < IC; c++) {
 176         for (idx_t i = 0; i < H; i++) {
 177           for (idx_t j = 0; j < W; j++) {
 178             max_value = max_i(max_value, rgb[c][i][j]);
 179           }
 180         }
 181       }
 182       float l_max = 1.0 / (float)max_value;
 183       dataset[k].index = k;
 184       dataset[k].label = label[0];
 185       for (idx_t c = 0; c < IC; c++) {
 186         for (idx_t i = 0; i < H; i++) {
 187           for (idx_t j = 0; j < W; j++) {
 188             dataset[k].rgb[c][i][j] = rgb[c][i][j];
 189             dataset[k](c,i,j) = rgb[c][i][j] * l_max;
 190           }
 191         }
 192       }
 193     }
 194     fclose(fp);
 195 
 196     if (dump_prefix) {
 197       dump_cifar_files(dataset, n_data, 0, dump_prefix);
 198     }
 199     
 200     /* shuffle data */
 201     rnd_gen_t rgv;
 202     rgv.seed(sample_seed);
 203     for (long t = 0; t < 15; t++) {
 204       for (long i = 0; i < n_data_in_file; i++) {
 205         long j = rgv.randi(i, n_data_in_file);
 206         cifar10_data_item<IC,H,W> d = dataset[j];
 207         dataset[j] = dataset[i];
 208         dataset[i] = d;
 209       }
 210     }
 211     train = dataset;
 212     validate = dataset + n_train;
 213     log_dataset(lgr);
 214     lgr.log(1, "loading data ends");
 215     return 1;
 216   }

 278   /**
 279      @brief load x and t with the a mini batch of B images
 280      @param (x) array to load images into
 281      @param (t) array to load true labels into
 282      @param (B) the number of images to pick
 283    */
 284   int get_data_train(array4<maxB,IC,H,W>& x, ivec<maxB>& t, ivec<maxB>& idxs, idx_t B) {
 285     assert(B <= maxB);
 286     x.set_n_rows(B);
 287     t.set_n(B);
 288     idxs.set_n(B);
 289     for (long b = 0; b < B; b++) {
 290       long idx = rg.randi(0, n_train);
 291       cifar10_data_item<IC,H,W>& itm = train[idx];
 292       idxs(b) = itm.index;
 293       t(b) = itm.label;
 294       for (idx_t ic = 0; ic < IC; ic++) {
 295         for (idx_t i = 0; i < H; i++) {
 296           for (idx_t j = 0; j < W; j++) {
 297             x(b,ic,i,j) = itm(ic,i,j);
 298           }
 299         }
 300       }
 301     }
 302     x.to_dev();
 303     t.to_dev();
 304     idxs.to_dev();
 305     return 1;
 306   }

 340    @brief entry point of this header file
 341    @param (argc) the number of command line args
 342    @param (argv) command line args
 343 */
 344 int cifar_main(int argc, char ** argv) {
     /*   */
 345   cmdline_opt opt = parse_args(argc, argv);
 346   const idx_t maxB = MAX_BATCH_SIZE;
 347   const idx_t B = opt.batch_sz;
 348   const idx_t C = 64;
 349   const idx_t H = 32;
 350   const idx_t W = 32;
 351   logger lgr;
 352   lgr.start_log(opt);
 353   cifar10_dataset<maxB,C,H,W> ds;
 354   ds.set_seed(opt.sample_seed);
 355   ds.get_n_data_in_file(opt.cifar_data);
 356   ds.load(lgr,
 357           opt.cifar_data,
 358           opt.partial_data, opt.partial_data_seed, opt.validate_ratio,
 359           opt.cifar_data_dump);
 360   array4<maxB,C,H,W> x;
 361   ivec<maxB> t;
 362   ivec<maxB> idxs;
 363   x.init_const(B, 0);
 364   t.init_const(B, 0);
 365   idxs.init_const(B, 0);
 366   ds.get_data_train(x, t, idxs, opt.batch_sz);
 367   lgr.end_log();
 368   return 0;
 369 }
