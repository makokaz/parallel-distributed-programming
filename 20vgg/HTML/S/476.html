 163 /**
 164    @brief global (x,y,z combined into an integer) thread ID
 165  */
 166 __device__ inline int get_thread_id() {
     /*   */
 167   int x = get_thread_id_x();
 168   int y = get_thread_id_y();
 169   int z = get_thread_id_z();
 170   int nx = get_nthreads_x();
 171   int ny = get_nthreads_y();
 172   return nx * ny * z + nx * y + x;
 173 }
