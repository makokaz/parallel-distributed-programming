518  real linear_grad_check_rand(cmdline_opt opt, logger * lgr, rnd_gen_t& rg, idx_t B) {
519  /* initialize linear parameters */
520  Linear<maxB,IC,nC> * linear = new Linear<maxB,IC,nC>();
521  linear->init(opt, lgr, rg);
522  linear->make_dev();
523  linear->to_dev();
524  /* make w - dw/2 and w + dw/2 */
525  Linear<maxB,IC,nC> * linear_minus = linear->copy();
526  Linear<maxB,IC,nC> * linear_plus = linear->copy();
527  /* make coefficients to make the single loss value */
528  array4<maxB,nC,1,1> * alpha = new array4<maxB,nC,1,1>();
529  alpha->make_dev(opt.gpu_algo);
530  alpha->init_uniform(B, rg, 0.0, 1.0);
531  alpha->to_dev();
532  /* make input (x) */
533  array4<maxB,IC,1,1> * x = new array4<maxB,IC,1,1>();
534  x->make_dev(opt.gpu_algo);
535  x->init_uniform(B, rg, 0.0, 1.0);
536  x->to_dev();
537  /* forward and backward */
538  array4<maxB,nC,1,1>& y = linear->forward(*x);
539  array4<maxB,IC,1,1>& gx = linear->backward(*alpha);
540  /* ensure the gradient is back to host */
541  linear->to_host();
542 
543  /* make dx */
544  real e = 1.0e-4;
545  array4<maxB,IC,1,1> * dx = new array4<maxB,IC,1,1>();
546  dx->init_uniform(B, rg, -e, e);
547  /* make x - dx/2 and x + dx/2 */
548  array4<maxB,IC,1,1> * x_minus = new array4<maxB,IC,1,1>(*x);
549  x_minus->make_dev(opt.gpu_algo);
550  array4<maxB,IC,1,1> * x_plus = new array4<maxB,IC,1,1>(*x);
551  x_plus->make_dev(opt.gpu_algo);
552  /* update on the host and send the to gpu */
553  x_minus->update(-0.5, *dx);
554  x_plus->update( 0.5, *dx);
555  x_minus->to_dev();
556  x_plus->to_dev();
557 
558  /* set gw to a random vector */
559  linear_minus->rand_grad(rg, -e, e);
560  linear_plus->set_grad(*linear_minus);
561  /* send them to gpu */
562  linear_minus->to_dev();
563  linear_plus->to_dev();
564  /* update weights using gw (update runs on gpu) */
565  linear_minus->update(-0.5); /* w -= dw/2 */
566  linear_plus->update(0.5); /* w += dw/2 */
567  /* make y(w-dw/2,x-dx/2), y(w+dw/2,x+dx/2) */
568  array4<maxB,nC,1,1>& y_minus = linear_minus->forward(*x_minus);
569  array4<maxB,nC,1,1>& y_plus = linear_plus->forward(*x_plus);
570  /* get the result back to host */
571  y_minus.to_host();
572  y_plus.to_host();
573 
574  /* get the single loss values */
575  real L_minus = alpha->dot(y_minus);
576  real L = alpha->dot(y);
577  real L_plus = alpha->dot(y_plus);
578  /* various inner products */
579  real gx_gx = gx.dot(gx); /* L/xL/x */
580  real dx_dx = dx->dot(*dx); /* L/xdx */
581  real gx_dx = gx.dot(*dx); /* dxdx */
582  real gw_gw = linear->gw_dot_gw(*linear); /* L/wL/w */
583  real dw_dw = linear_minus->gw_dot_gw(*linear_minus); /* L/wdw */
584  real gw_dw = linear->gw_dot_gw(*linear_minus); /* dwdw */
585 
586  real rel_e = show_error(gx_gx, dx_dx, gx_dx, gw_gw, dw_dw, gw_dw, L_minus, L, L_plus);
587  /* clean up */
588  linear->del_dev();
589  linear_minus->del_dev();
590  linear_plus->del_dev();
591  alpha->del_dev();
592  x->del_dev();
593  dx->del_dev();
594  x_minus->del_dev();
595  x_plus->del_dev();
596 
597  delete linear;
598  delete linear_minus;
599  delete linear_plus;
600  delete alpha;
601  delete x;
602  delete dx;
603  delete x_minus;
604  delete x_plus;
605  return rel_e;
