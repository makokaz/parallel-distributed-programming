403  real maxpooling_grad_check_rand(cmdline_opt opt, logger * lgr, rnd_gen_t& rg, idx_t B) {
404  /* initialize maxpooling parameters */

406  mp->init(opt, lgr);
407  mp->make_dev();
408  mp->to_dev();
409 
410  /* make copies */
411  MaxPooling2D<maxB,C,H,W,S> * mp_minus = mp->copy();
412  MaxPooling2D<maxB,C,H,W,S> * mp_plus = mp->copy();
413  /* make coefficients to make the single loss value */
414  array4<maxB,C,H/S,W/S> * alpha = new array4<maxB,C,H/S,W/S>();
415  alpha->make_dev(opt.gpu_algo);
416  alpha->init_uniform(B, rg, 0.0, 1.0);
417  alpha->to_dev();
418  /* make input (x) */
419  array4<maxB,C,H,W> * x = new array4<maxB,C,H,W>();
420  x->make_dev(opt.gpu_algo);
421  x->init_uniform(B, rg, 0.0, 1.0);
422  x->to_dev();
423  /* forward and backward */
424  array4<maxB,C,H/S,W/S>& y = mp->forward(*x);
425  array4<maxB,C,H,W>& gx = mp->backward(*alpha);
426  mp->to_host();
427 
428  /* make dx */
429  real e = 1.0e-4;
430  array4<maxB,C,H,W> * dx = new array4<maxB,C,H,W>();
431  dx->init_uniform(B, rg, -e, e);
432  /* make x - dx/2 and x + dx/2 */
433  array4<maxB,C,H,W> * x_minus = new array4<maxB,C,H,W>(*x);
434  x_minus->make_dev(opt.gpu_algo);
435  array4<maxB,C,H,W> * x_plus = new array4<maxB,C,H,W>(*x);
436  x_plus->make_dev(opt.gpu_algo);
437  x_minus->update(-0.5, *dx);
438  x_plus->update( 0.5, *dx);
439  x_minus->to_dev();
440  x_plus->to_dev();
441 
442  /* send copies to gpu */
443  mp_minus->to_dev();
444  mp_plus->to_dev();
445  /* make y(x-dx/2), y(x+dx/2) */
446  array4<maxB,C,H/S,W/S>& y_minus = mp_minus->forward(*x_minus);
447  array4<maxB,C,H/S,W/S>& y_plus = mp_plus->forward(*x_plus);
448  /* get the result back to host */
449  y_minus.to_host();
450  y_plus.to_host();
451 
452  /* get the single loss values */
