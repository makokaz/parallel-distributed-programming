550  static real convolution_grad_check_rand(cmdline_opt opt, logger * lgr, rnd_gen_t& rg, idx_t B) {
551  /* make weight and transfer to gpu if working on gpu */
552  Convolution2D<maxB,IC,H,W,K,OC> * conv = new Convolution2D<maxB,IC,H,W,K,OC>();
553  conv->init(opt, lgr, rg);
554  conv->make_dev();
555  conv->to_dev();
556  /* make w - dw/2 and w + dw/2 */
557  Convolution2D<maxB,IC,H,W,K,OC> * conv_minus = conv->copy();
558  Convolution2D<maxB,IC,H,W,K,OC> * conv_plus = conv->copy();
559  /* make coefficients to make the single loss value */
560  array4<maxB,OC,H,W> * alpha = new array4<maxB,OC,H,W>();
561  alpha->make_dev(opt.gpu_algo);
562  alpha->init_uniform(B, rg, 0.0, 1.0);
563  alpha->to_dev();
564  /* make input (x) */
565  array4<maxB,IC,H,W> * x = new array4<maxB,IC,H,W>();
566  x->make_dev(opt.gpu_algo);
567  x->init_uniform(B, rg, 0.0, 1.0);
568  x->to_dev();
569  /* forward and backward */
570  array4<maxB,OC,H,W>& y = conv->forward(*x);
571  array4<maxB,IC,H,W>& gx = conv->backward(*alpha);
572  /* ensure the gradient is back to host */
573  conv->to_host();
574 
575  /* make dx */
576  real e = 1.0e-4;
577  array4<maxB,IC,H,W> * dx = new array4<maxB,IC,H,W>();
578  dx->init_uniform(B, rg, -e, e);
579  /* make x - dx/2 and x + dx/2 */
580  array4<maxB,IC,H,W> * x_minus = new array4<maxB,IC,H,W>(*x);
581  x_minus->make_dev(opt.gpu_algo);
582  array4<maxB,IC,H,W> * x_plus = new array4<maxB,IC,H,W>(*x);
583  x_plus->make_dev(opt.gpu_algo);
584  /* update on the host and send the to gpu */
585  x_minus->update(-0.5, *dx);
586  x_plus->update( 0.5, *dx);
587  x_minus->to_dev();
588  x_plus->to_dev();
589 
590  /* set gw to a random vector */
591  conv_minus->rand_grad(rg, -e, e);
592  conv_plus->set_grad(*conv_minus);
593  /* send them to gpu */
594  conv_minus->to_dev();
595  conv_plus->to_dev();
596  /* update weights using gw (update runs on gpu) */
597  conv_minus->update(-0.5); /* w -= dw/2 */
598  conv_plus->update(0.5); /* w += dw/2 */
599  /* make y(w-dw/2,x-dx/2), y(w+dw/2,x+dx/2) */
600  array4<maxB,OC,H,W>& y_minus = conv_minus->forward(*x_minus);
601  array4<maxB,OC,H,W>& y_plus = conv_plus->forward(*x_plus);
602  /* get the result back to host */
603  y_minus.to_host();
604  y_plus.to_host();
605 
606  /* get the single loss values */
607  real L_minus = alpha->dot(y_minus);
608  real L = alpha->dot(y);
609  real L_plus = alpha->dot(y_plus);
610  /* various inner products */
611  real gx_gx = gx.dot(gx); /* ∂L/∂x・∂L/∂x */
612  real dx_dx = dx->dot(*dx); /* ∂L/∂x・dx */
613  real gx_dx = gx.dot(*dx); /* dx・dx */
614  real gw_gw = conv->gw_dot_gw(*conv); /* ∂L/∂w・∂L/∂w */
615  real dw_dw = conv_minus->gw_dot_gw(*conv_minus); /* ∂L/∂w・dw */
616  real gw_dw = conv->gw_dot_gw(*conv_minus); /* dw・dw */
617 
618  real rel_e = show_error(gx_gx, dx_dx, gx_dx, gw_gw, dw_dw, gw_dw, L_minus, L, L_plus);
619  /* clean up */
620  conv->del_dev();
621  conv_minus->del_dev();
622  conv_plus->del_dev();
623  alpha->del_dev();
624  x->del_dev();
625  dx->del_dev();
626  x_minus->del_dev();
627  x_plus->del_dev();
628 
629  delete conv;
630  delete conv_minus;
631  delete conv_plus;
632  delete alpha;
633  delete x;
634  delete dx;
635  delete x_minus;
636  delete x_plus;
637  return rel_e;
638 }
