437  __device__ __host__
438  double rand01() {
439  union ieee754_double temp;
440  /* Compute next state. */
441  next();
442  /* Construct a positive double with the 48 random bits distributed over
443  its fractional part so the resulting FP number is [0.0,1.0). */
444  temp.ieee.negative = 0;
445  temp.ieee.exponent = IEEE754_DOUBLE_BIAS;
446  temp.ieee.mantissa0 = (x >> 28) & ((1UL << 20) - 1); /* 20 bit */
447  temp.ieee.mantissa1 = (x & ((1UL << 28) - 1)) << 4;
448  /* Please note the lower 4 bits of mantissa1 are always 0. */
449  return temp.d - 1.0;
450  }
454  __device__ __host__
455  long randi32() {
456  /* Compute next state. */
457  next();
458  /* Store the result. */
459  return (x >> 17) & ((1UL << 31) - 1);
460  }

517 static real show_error(real gx_gx, real dx_dx, real gx_dx,
518  real gw_gw, real dw_dw, real gw_dw,
519  real L_minus, real L, real L_plus) {
520  printf("|L/x| = %.9f\n", sqrt(gx_gx));
521  printf("|dx| = %.9f\n", sqrt(dx_dx));
522  printf("L/xdx = %.9f\n", gx_dx);
523  printf("|L/w| = %.9f\n", sqrt(gw_gw));
524  printf("|dw| = %.9f\n", sqrt(dw_dw));
525  printf("L/wdw = %.9f\n", gw_dw);
526  printf("L- = %.9f\n", L_minus);
527  printf("L = %.9f\n", L);
528  printf("L+ = %.9f\n", L_plus);
529  real dL = L_plus - L_minus;
530  real A = gx_dx + gw_dw;
531  real B = dL;
532  real e = (A == B ? 0.0 : fabs(A - B) / max_r(fabs(A), fabs(B)));
533  printf("A = L/xdx + L/wdw = %.9f\n", gx_dx + gw_dw);
534  printf("B = L = %.9f\n", dL);
535  printf("relative error = |A-B|/max(|A|,|B|) = %.9f\n", e);
536  return e;
537 }
