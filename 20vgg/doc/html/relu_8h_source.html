382  static real relu_grad_check_rand(cmdline_opt opt, logger * lgr, rnd_gen_t& rg, idx_t B) {
383  /* initialize relu parameters */
384  Relu<maxB,C,H,W> * relu = new Relu<maxB,C,H,W>();
385  relu->init(opt, lgr);
386  relu->make_dev();
387  relu->to_dev();
388  /* make copies */
389  Relu<maxB,C,H,W> * relu_minus = relu->copy();
390  Relu<maxB,C,H,W> * relu_plus = relu->copy();
391  /* make coefficients to make the single loss value */
392  array4<maxB,C,H,W> * alpha = new array4<maxB,C,H,W>();
393  alpha->make_dev(opt.gpu_algo);
394  alpha->init_uniform(B, rg, 0.0, 1.0);
395  alpha->to_dev();
396  /* make input (x) */
397  array4<maxB,C,H,W> * x = new array4<maxB,C,H,W>();
398  x->make_dev(opt.gpu_algo);
399  x->init_uniform(B, rg, 0.0, 1.0);
400  x->to_dev();
401  /* forward and backward */
402  array4<maxB,C,H,W>& y = relu->forward(*x);
403  array4<maxB,C,H,W>& gx = relu->backward(*alpha);
404  relu->to_host();
405  /* make dx */
406  real e = 1.0e-4;
407  array4<maxB,C,H,W> * dx = new array4<maxB,C,H,W>();
408  dx->init_uniform(B, rg, -e, e);
409  /* make x - dx/2 and x + dx/2 */
410  array4<maxB,C,H,W> * x_minus = new array4<maxB,C,H,W>(*x);
411  x_minus->make_dev(opt.gpu_algo);
412  array4<maxB,C,H,W> * x_plus = new array4<maxB,C,H,W>(*x);
413  x_plus->make_dev(opt.gpu_algo);
414  x_minus->update(-0.5, *dx);
415  x_plus->update(0.5, *dx);
416  x_minus->to_dev();
417  x_plus->to_dev();
418 
419  /* send copies to gpu */
420  relu_minus->to_dev();
421  relu_plus->to_dev();
422  /* make y(x-dx/2), y(x+dx/2) */
423  array4<maxB,C,H,W>& y_minus = relu_minus->forward(*x_minus);
424  array4<maxB,C,H,W>& y_plus = relu_plus->forward(*x_plus);
425  /* get the result back to host */
426  y_minus.to_host();
427  y_plus.to_host();
428 
429  /* get the single loss values */
430  real L_minus = alpha->dot(y_minus);
431  real L = alpha->dot(y);
432  real L_plus = alpha->dot(y_plus);
433  /* various inner products */
434  real gx_gx = gx.dot(gx); /* L/xL/x */
435  real dx_dx = dx->dot(*dx); /* L/xdx */
436  real gx_dx = gx.dot(*dx); /* dxdx */
437  real gw_gw = 0; /* L/wL/w */
438  real dw_dw = 0; /* L/wdw */
439  real gw_dw = 0; /* dwdw */
440 
441  real rel_e = show_error(gx_gx, dx_dx, gx_dx, gw_gw, dw_dw, gw_dw, L_minus, L, L_plus);
442  /* clean up */
443  relu->del_dev();
444  relu_minus->del_dev();
445  relu_plus->del_dev();
446  alpha->del_dev();
447  x->del_dev();
448  dx->del_dev();
449  x_minus->del_dev();
450  x_plus->del_dev();
