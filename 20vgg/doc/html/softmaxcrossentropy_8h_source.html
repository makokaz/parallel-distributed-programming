424  static real softmaxcrossentropy_grad_check_rand(cmdline_opt opt, logger * lgr, rnd_gen_t& rg, idx_t B) {
425  /* initialize softmax */

427  smxe->init(opt, lgr);
428  smxe->make_dev();
429  smxe->to_dev();
430  /* make copies */
431  SoftmaxCrossEntropy<maxB,nC>* smxe_minus = smxe->copy();
432  SoftmaxCrossEntropy<maxB,nC>* smxe_plus = smxe->copy();
433  /* make coefficients to make the single loss value */
434  vec<maxB> * alpha = new vec<maxB>();
435  alpha->make_dev(opt.gpu_algo);
436  alpha->init_uniform(B, rg, 0.0, 1.0);
437  alpha->to_dev();
438  /* make input (x) */
439  array4<maxB,nC,1,1> * x = new array4<maxB,nC,1,1>();
440  x->make_dev(opt.gpu_algo);
441  x->init_uniform(B, rg, 0.0, 1.0);
442  x->to_dev();
443  /* make input (t) */
444  ivec<maxB> * t = new ivec<maxB>();
445  t->make_dev(opt.gpu_algo);
446  t->init_uniform(B, rg, 0, nC);
447  t->to_dev();
448  /* forward and backward */
449  vec<maxB>& y = smxe->forward(*x, *t);
450  array4<maxB,nC,1,1>& gx = smxe->backward(*alpha);
451  smxe->to_host();
452 
453  /* make dx */
454  real e = 1.0e-4;
455  array4<maxB,nC,1,1> * dx = new array4<maxB,nC,1,1>();
456  dx->init_uniform(B, rg, -e, e);
457  /* make x - dx/2 and x + dx/2 */
458  array4<maxB,nC,1,1> * x_minus = new array4<maxB,nC,1,1>(*x);
459  x_minus->make_dev(opt.gpu_algo);
460  array4<maxB,nC,1,1> * x_plus = new array4<maxB,nC,1,1>(*x);
461  x_plus->make_dev(opt.gpu_algo);
462  x_minus->update(-0.5, *dx);
463  x_plus->update( 0.5, *dx);
464  x_minus->to_dev();
465  x_plus->to_dev();
466 
467  /* send copies to gpu */
468  smxe_minus->to_dev();
469  smxe_plus->to_dev();
470  /* make y(x-dx/2), y(x+dx/2) */
471  vec<maxB>& y_minus = smxe_minus->forward(*x_minus, *t);
472  vec<maxB>& y_plus = smxe_plus->forward(*x_plus, *t);
473  /* get the result back to host */
474  y_minus.to_host();
475  y_plus.to_host();
476 
477  /* get the single loss values */

479  real L = alpha->dot(y);
480  real L_plus = alpha->dot(y_plus);
481  /* various inner products */

483  real dx_dx = dx->dot(*dx); /* ∂L/∂x・dx */
