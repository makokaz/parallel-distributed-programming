245  static real block_grad_check_rand(cmdline_opt opt, logger * lgr, rnd_gen_t& rg, idx_t B) {
246  /* initialize block parameters */
247  Block<maxB,IC,H,W,K,OC> * block = new Block<maxB,IC,H,W,K,OC>();
248  block->init(opt, lgr, rg);
249  block->make_dev();
250  block->to_dev();
251  /* make w - dw/2 and w + dw/2 */
252  Block<maxB,IC,H,W,K,OC> * block_minus = block->copy();
253  Block<maxB,IC,H,W,K,OC> * block_plus = block->copy();
254  /* make coefficients to make the single loss value */
255  array4<maxB,OC,H,W> * alpha = new array4<maxB,OC,H,W>();
256  alpha->make_dev(opt.gpu_algo);
257  alpha->init_uniform(B, rg, 0.0, 1.0);
258  alpha->to_dev();
259  /* make input (x) */
260  array4<maxB,IC,H,W> * x = new array4<maxB,IC,H,W>();
261  x->make_dev(opt.gpu_algo);
262  x->init_uniform(B, rg, 0.0, 1.0);
263  x->to_dev();
264  /* forward and backward */
265  array4<maxB,OC,H,W>& y = block->forward(*x);
266  array4<maxB,IC,H,W>& gx = block->backward(*alpha);
267  /* ensure the gradient is back to host */
268  block->to_host();
269 
270  /* make dx */
271  real e = 1.0e-4;
272  array4<maxB,IC,H,W> * dx = new array4<maxB,IC,H,W>();
273  dx->init_uniform(B, rg, -e, e);
274  /* make x - dx/2 and x + dx/2 */
275  array4<maxB,IC,H,W> * x_minus = new array4<maxB,IC,H,W>(*x);
276  x_minus->make_dev(opt.gpu_algo);
277  array4<maxB,IC,H,W> * x_plus = new array4<maxB,IC,H,W>(*x);
278  x_plus->make_dev(opt.gpu_algo);
279  /* update on the host and send the to gpu */
280  x_minus->update(-0.5, *dx);
281  x_plus->update( 0.5, *dx);
282  x_minus->to_dev();
283  x_plus->to_dev();
284 
285  /* set gw to a random vector */
286  block_minus->rand_grad(rg, -e, e);
287  block_plus->set_grad(*block_minus);
288  /* send them to gpu */
289  block_minus->to_dev();
290  block_plus->to_dev();
291  /* update weights using gw (update runs on gpu) */
292  block_minus->update(-0.5); /* w -= dw/2 */
293  block_plus->update(0.5); /* w += dw/2 */
294  /* make y(w-dw/2,x-dx/2), y(w+dw/2,x+dx/2) */
295  array4<maxB,OC,H,W>& y_minus = block_minus->forward(*x_minus);
296  array4<maxB,OC,H,W>& y_plus = block_plus->forward(*x_plus);
297  /* get the result back to host */
298  y_minus.to_host();
299  y_plus.to_host();
300 
301  /* get the single loss values */
302  real L_minus = alpha->dot(y_minus);
303  real L = alpha->dot(y);
304  real L_plus = alpha->dot(y_plus);
305  /* various inner products */
306  real gx_gx = gx.dot(gx); /* ∂L/∂x・∂L/∂x */
307  real dx_dx = dx->dot(*dx); /* ∂L/∂x・dx */
308  real gx_dx = gx.dot(*dx); /* dx・dx */
309  real gw_gw = block->gw_dot_gw(*block); /* ∂L/∂w・∂L/∂w */
310  real dw_dw = block_minus->gw_dot_gw(*block_minus); /* ∂L/∂w・dw */
311  real gw_dw = block->gw_dot_gw(*block_minus); /* dw・dw */
312 
313  real rel_e = show_error(gx_gx, dx_dx, gx_dx, gw_gw, dw_dw, gw_dw, L_minus, L, L_plus);
314  /* clean up */
315  block->del_dev();
316  block_minus->del_dev();
317  block_plus->del_dev();
318  alpha->del_dev();
319  x->del_dev();
320  dx->del_dev();
321  x_minus->del_dev();
322  x_plus->del_dev();
323 
324  delete block;
325  delete block_minus;
326  delete block_plus;
327  delete alpha;
328  delete x;
329  delete dx;
330  delete x_minus;
331  delete x_plus;
332  return rel_e;
333 }
