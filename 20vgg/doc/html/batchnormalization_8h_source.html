287  const idx_t B = x.B;

315  const idx_t B = x.B;

317  const real l_BHW = 1 / (real)(B * H * W);
318  inv_std.set_n(IC);
319  for (idx_t ic = 0; ic < IC; ic++) {
320  real s = 0.0;
321  for (idx_t b = 0; b < B; b++) {
322  for (idx_t i = 0; i < H; i++) {
323  for (idx_t j = 0; j < W; j++) {
324  real ds = x(b,ic,i,j) - mu(ic);
325  s += ds * ds;
326  }
327  }
328  }
329  inv_std(ic) = 1.0 / sqrt(s * l_BHW + epsilon);

631  static real batchnormalization_grad_check_rand(cmdline_opt opt, logger * lgr, rnd_gen_t& rg, idx_t B) {
632  /* initialize batch normalization parameters */
633  BatchNormalization<maxB,IC,H,W> * bn = new BatchNormalization<maxB,IC,H,W>();
634  bn->init(opt, lgr, rg);
635  bn->make_dev();
636  bn->to_dev();
637  /* make w - dw/2 and w + dw/2 */
638  BatchNormalization<maxB,IC,H,W> * bn_minus = bn->copy();
639  BatchNormalization<maxB,IC,H,W> * bn_plus = bn->copy();
640  /* make coefficients to make the single loss value */
641  array4<maxB,IC,H,W> * alpha = new array4<maxB,IC,H,W>();
642  alpha->make_dev(opt.gpu_algo);
643  alpha->init_uniform(B, rg, 0.0, 1.0);
644  alpha->to_dev();
645  /* make input (x) */
646  array4<maxB,IC,H,W> * x = new array4<maxB,IC,H,W>();
647  x->make_dev(opt.gpu_algo);
648  x->init_uniform(B, rg, 0.0, 1.0);
649  x->to_dev();
650  /* forward and backward */
651  array4<maxB,IC,H,W>& y = bn->forward(*x);
652  array4<maxB,IC,H,W>& gx = bn->backward(*alpha);
653  /* ensure the gradient is back to host */
654  bn->to_host();
655 
656  /* make dx */
657  real e = 1.0e-4;
658  array4<maxB,IC,H,W> * dx = new array4<maxB,IC,H,W>();
659  dx->init_uniform(B, rg, -e, e);
660  /* make x - dx/2 and x + dx/2 */
661  array4<maxB,IC,H,W> * x_minus = new array4<maxB,IC,H,W>(*x);
662  x_minus->make_dev(opt.gpu_algo);
663  array4<maxB,IC,H,W> * x_plus = new array4<maxB,IC,H,W>(*x);
664  x_plus->make_dev(opt.gpu_algo);
665  /* update on the host and send the to gpu */
666  x_minus->update(-0.5, *dx);
667  x_plus->update( 0.5, *dx);
668  x_minus->to_dev();
669  x_plus->to_dev();
670 
671  /* set gw to a random vector */
672  bn_minus->rand_grad(rg, -e, e);
673  bn_plus->set_grad(*bn_minus);
674  /* send them to gpu */
675  bn_minus->to_dev();
676  bn_plus->to_dev();
677  /* update weights using gw (update runs on gpu) */
678  bn_minus->update(-0.5); /* w -= dw/2 */
679  bn_plus->update(0.5); /* w += dw/2 */
680  /* make y(w-dw,x-dx), y(w+dw,x+dx) */
681  array4<maxB,IC,H,W>& y_minus = bn_minus->forward(*x_minus);
682  array4<maxB,IC,H,W>& y_plus = bn_plus->forward(*x_plus);
683  /* get the result back to host */
684  y_minus.to_host();
685  y_plus.to_host();
686 
687  /* get the single loss values */
688  real L_minus = alpha->dot(y_minus);
689  real L = alpha->dot(y);
690  real L_plus = alpha->dot(y_plus);
691  /* various inner products */
692  real gx_gx = gx.dot(gx); /* L/xL/x */
693  real dx_dx = dx->dot(*dx); /* L/xdx */
694  real gx_dx = gx.dot(*dx); /* dxdx */
695  real gw_gw = bn->gw_dot_gw(*bn); /* L/wL/w */
696  real dw_dw = bn_minus->gw_dot_gw(*bn_minus); /* L/wdw */
697  real gw_dw = bn->gw_dot_gw(*bn_minus); /* dwdw */
698 
699  real rel_e = show_error(gx_gx, dx_dx, gx_dx, gw_gw, dw_dw, gw_dw, L_minus, L, L_plus);
700  /* clean up */
701  bn_minus->del_dev();
702  bn_plus->del_dev();
703  alpha->del_dev();
704  x->del_dev();
705  dx->del_dev();
706  x_minus->del_dev();
707  x_plus->del_dev();
708 
709  delete bn_minus;
710  delete bn_plus;
711  delete alpha;
712  delete x;
713  delete dx;
714  delete x_minus;
715  delete x_plus;
716  return rel_e;
717 }
