424  real forward_backward_update(array4<maxB,C0,H,W>& x, ivec<maxB>& t, real eta) {
425  const idx_t B = x.B;
426  /* forward */
427  vec<maxB>& y = forward(x, t);
428  /* a vector (1,1,1,...) to make the single loss value from loss of each sample */
429  gy.init_const(B, 1.0);
430  gy.to_dev();
431  /* backward (set weights of all sublayers) */
432  backward(gy);
433  /* update */
434  real e = eta / B;
435  update(-e);
436  /* get the loss of each sample back to host if we are working on GPU */
437  y.to_host();
438  real L = gy.dot(y);
439  return L;
440  }

552  vgg->init(opt, lgr, rg);
553  vgg->make_dev();
554  vgg->to_dev();
555  /* make w - dw/2 and w + dw/2 */
556  VGG<maxB,C0,H,W,K,S,C1,nC> * vgg_minus = vgg->copy();
557  VGG<maxB,C0,H,W,K,S,C1,nC> * vgg_plus = vgg->copy();
558  /* make coefficients to make the single loss value */
559  vec<maxB> * alpha = new vec<maxB>();
560  alpha->make_dev(opt.gpu_algo);
561  alpha->init_uniform(B, rg, 0.0, 1.0);
562  alpha->to_dev();
563  /* make input (x) */
564  array4<maxB,C0,H,W> * x = new array4<maxB,C0,H,W>();
565  x->make_dev(opt.gpu_algo);
566  x->init_uniform(B, rg, 0.0, 1.0);
567  x->to_dev();
568  /* make input (t) */
569  ivec<maxB>* t = new ivec<maxB>();
570  t->make_dev(opt.gpu_algo);
571  t->init_uniform(B, rg, 0, nC);
572  t->to_dev();
573  /* forward and backward */
574  vec<maxB>& y = vgg->forward(*x, *t);
575  array4<maxB,C0,H,W>& gx = vgg->backward(*alpha);
576  /* ensure the gradient is back to host */
577  vgg->to_host();
578 
579  /* make dx */
580  real e = 1.0e-4;
581  array4<maxB,C0,H,W> * dx = new array4<maxB,C0,H,W>();
582  dx->init_uniform(B, rg, -e, e);
583  /* make x - dx/2 and x + dx/2 */
584  array4<maxB,C0,H,W> * x_minus = new array4<maxB,C0,H,W>(*x);
585  x_minus->make_dev(opt.gpu_algo);
586  array4<maxB,C0,H,W> * x_plus = new array4<maxB,C0,H,W>(*x);
587  x_plus->make_dev(opt.gpu_algo);
588  /* update on the host and send the to gpu */
589  x_minus->update(-0.5, *dx);
590  x_plus->update( 0.5, *dx);
591  x_minus->to_dev();
592  x_plus->to_dev();
593 
594  /* set gw to a random vector */
595  vgg_minus->rand_grad(rg, -e, e);
596  vgg_plus->set_grad(*vgg_minus);
597  /* send them to gpu */
598  vgg_minus->to_dev();
599  vgg_plus->to_dev();
600  /* update weights using gw (update runs on gpu) */
601  vgg_minus->update(-0.5); /* w -= dw/2 */
602  vgg_plus->update(0.5); /* w += dw/2 */
603  /* make y(w-dw/2,x-dx/2), y(w+dw/2,x+dx/2) */
604  vec<maxB>& y_minus = vgg_minus->forward(*x_minus, *t);
605  vec<maxB>& y_plus = vgg_plus->forward(*x_plus, *t);
606  /* get the result back to host */
607  y_minus.to_host();
608  y_plus.to_host();
609 
610  /* get the single loss values */
611  real L_minus = alpha->dot(y_minus);
612  real L = alpha->dot(y);
613  real L_plus = alpha->dot(y_plus);
614  /* various inner products */
615  real gx_gx = gx.dot(gx); /* L/xL/x */
616  real dx_dx = dx->dot(*dx); /* L/xdx */
617  real gx_dx = gx.dot(*dx); /* dxdx */
618  real gw_gw = vgg->gw_dot_gw(*vgg); /* L/wL/w */
619  real dw_dw = vgg_minus->gw_dot_gw(*vgg_minus); /* L/wdw */
620  real gw_dw = vgg->gw_dot_gw(*vgg_minus); /* dwdw */
621 
622  real rel_e = show_error(gx_gx, dx_dx, gx_dx, gw_gw, dw_dw, gw_dw, L_minus, L, L_plus);
623  /* clean up */
624  vgg->del_dev();
625  vgg_minus->del_dev();
626  vgg_plus->del_dev();
627  alpha->del_dev();
628  x->del_dev();
629  dx->del_dev();
630  x_minus->del_dev();
631  x_plus->del_dev();
632 
633  delete vgg;
634  delete vgg_minus;
635  delete vgg_plus;
636  delete alpha;
637  delete x;
638  delete dx;
639  delete x_minus;
640  delete x_plus;
641  return rel_e;
642 }
