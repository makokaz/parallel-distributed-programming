401  static real dropout_grad_check_rand(cmdline_opt opt, logger * lgr, rnd_gen_t& rg, idx_t B) {
402  /* initialize dropout parameters */
403  const real drop_ratio = 0.3;
404  Dropout<maxB,C,H,W> * dropout = new Dropout<maxB,C,H,W>();
405  dropout->init(opt, lgr, drop_ratio, opt.dropout_seed);
406  dropout->make_dev();
407  dropout->to_dev();
408  /* make copies */
409  Dropout<maxB,C,H,W> * dropout_minus = dropout->copy();
410  Dropout<maxB,C,H,W> * dropout_plus = dropout->copy();
411  /* make coefficients to make the single loss value */
412  array4<maxB,C,H,W> * alpha = new array4<maxB,C,H,W>();
413  alpha->make_dev(opt.gpu_algo);
414  alpha->init_uniform(B, rg, 0.0, 1.0);
415  alpha->to_dev();
416  /* make input (x) */
417  array4<maxB,C,H,W> * x = new array4<maxB,C,H,W>();
418  x->make_dev(opt.gpu_algo);
419  x->init_uniform(B, rg, 0.0, 1.0);
420  x->to_dev();
421  /* forward and backward */
422  array4<maxB,C,W,H>& y = dropout->forward(*x);
423  array4<maxB,C,H,W>& gx = dropout->backward(*alpha);
424  dropout->to_host();
425 
426  /* make dx */
427  real e = 1.0e-4;
428  array4<maxB,C,H,W> * dx = new array4<maxB,C,H,W>();
429  dx->init_uniform(B, rg, -e, e);
430  /* make x - dx/2 and x + dx/2 */
431  array4<maxB,C,H,W> * x_minus = new array4<maxB,C,H,W>(*x);
432  x_minus->make_dev(opt.gpu_algo);
433  array4<maxB,C,H,W> * x_plus = new array4<maxB,C,H,W>(*x);
434  x_plus->make_dev(opt.gpu_algo);
435  x_minus->update(-0.5, *dx);
436  x_plus->update( 0.5, *dx);
437  x_minus->to_dev();
438  x_plus->to_dev();
439 
440  /* send copies to gpu */
441  dropout_minus->to_dev();
442  dropout_plus->to_dev();
443  /* make y(x-dx/2), y(x+dx/2) */
444  array4<maxB,C,H,W>& y_minus = dropout_minus->forward(*x_minus);
445  array4<maxB,C,H,W>& y_plus = dropout_plus->forward(*x_plus);
446  /* get the result back to host */
447  y_minus.to_host();
448  y_plus.to_host();
449 
450  /* get the single loss values */
451  real L_minus = alpha->dot(y_minus);
452  real L = alpha->dot(y);
453  real L_plus = alpha->dot(y_plus);
454  /* various inner products */
455  real gx_gx = gx.dot(gx); /* ∂L/∂x・∂L/∂x */
456  real dx_dx = dx->dot(*dx); /* ∂L/∂x・dx */
457  real gx_dx = gx.dot(*dx); /* dx・dx */
458  real gw_gw = 0; /* ∂L/∂w・∂L/∂w */
459  real dw_dw = 0; /* ∂L/∂w・dw */
460  real gw_dw = 0; /* dw・dw */
461 
462  real rel_e = show_error(gx_gx, dx_dx, gx_dx, gw_gw, dw_dw, gw_dw, L_minus, L, L_plus);
463  /* clean up */
464  dropout->del_dev();
465  dropout_minus->del_dev();
466  dropout_plus->del_dev();
467  alpha->del_dev();
468  x->del_dev();
469  dx->del_dev();
470  x_minus->del_dev();
471  x_plus->del_dev();
472 
473  delete dropout;
474  delete dropout_minus;
475  delete dropout_plus;
476  delete alpha;
477  delete x;
478  delete dx;
479  delete x_minus;
480  delete x_plus;
481  return rel_e;
482 }
